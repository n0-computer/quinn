46/// A QUIC endpoint.
47///
48/// An endpoint corresponds to a single UDP socket, may host many connections, and may act as both
49/// client and server for different connections.
50///
51/// May be cloned to obtain another handle to the same endpoint.
52#[derive(Debug, Clone)]
53pub struct Endpoint {
54    pub(crate) inner: EndpointRef,
55    runtime: Arc<dyn Runtime>,
56}

58impl Endpoint {
59    /// Helper to construct an endpoint for use with outgoing connections only
60    ///
61    /// Note that `addr` is the *local* address to bind to, which should usually be a wildcard
62    /// address like `0.0.0.0:0` or `[::]:0`, which allow communication with any reachable IPv4 or
63    /// IPv6 address respectively from an OS-assigned port.
64    ///
65    /// If an IPv6 address is provided, attempts to make the socket dual-stack so as to allow
66    /// communication with both IPv4 and IPv6 addresses. As such, calling `Endpoint::client` with
67    /// the address `[::]:0` is a reasonable default to maximize the ability to connect to other
68    /// address. For example:
69    ///
70    /// ```
71    /// iroh_quinn::Endpoint::client((std::net::Ipv6Addr::UNSPECIFIED, 0).into());
72    /// ```
73    ///
74    /// Some environments may not allow creation of dual-stack sockets, in which case an IPv6
75    /// client will only be able to connect to IPv6 servers. An IPv4 client is never dual-stack.
76    #[cfg(all(
77        not(wasm_browser),
78        any(feature = "runtime-tokio", feature = "runtime-smol"),
79        any(feature = "aws-lc-rs", feature = "ring"), // `EndpointConfig::default()` is only available with these
80    ))]
81    pub fn client(addr: SocketAddr) -> io::Result<Self> {
82        let socket = Socket::new(Domain::for_address(addr), Type::DGRAM, Some(Protocol::UDP))?;
83        if addr.is_ipv6()
84            && let Err(e) = socket.set_only_v6(false)
85        {
86            tracing::debug!(%e, "unable to make socket dual-stack");
87        }
88        socket.bind(&addr.into())?;
89        let runtime =
90            default_runtime().ok_or_else(|| io::Error::other("no async runtime found"))?;
91        Self::new_with_abstract_socket(
92            EndpointConfig::default(),
93            None,
94            runtime.wrap_udp_socket(socket.into())?,
95            runtime,
96        )
97    }

104    /// Helper to construct an endpoint for use with both incoming and outgoing connections
105    ///
106    /// Platform defaults for dual-stack sockets vary. For example, any socket bound to a wildcard
107    /// IPv6 address on Windows will not by default be able to communicate with IPv4
108    /// addresses. Portable applications should bind an address that matches the family they wish to
109    /// communicate within.
110    #[cfg(all(
111        not(wasm_browser),
112        any(feature = "runtime-tokio", feature = "runtime-smol"),
113        any(feature = "aws-lc-rs", feature = "ring"), // `EndpointConfig::default()` is only available with these
114    ))]
115    pub fn server(config: ServerConfig, addr: SocketAddr) -> io::Result<Self> {
116        let socket = std::net::UdpSocket::bind(addr)?;
117        let runtime =
118            default_runtime().ok_or_else(|| io::Error::other("no async runtime found"))?;
119        Self::new_with_abstract_socket(
120            EndpointConfig::default(),
121            Some(config),
122            runtime.wrap_udp_socket(socket)?,
123            runtime,
124        )
125    }

139    /// Construct an endpoint with arbitrary configuration and pre-constructed abstract socket
140    ///
141    /// Useful when `socket` has additional state (e.g. sidechannels) attached for which shared
142    /// ownership is needed.
143    pub fn new_with_abstract_socket(
144        config: EndpointConfig,
145        server_config: Option<ServerConfig>,
146        socket: Box<dyn AsyncUdpSocket>,
147        runtime: Arc<dyn Runtime>,
148    ) -> io::Result<Self> {
149        let addr = socket.local_addr()?;
150        let allow_mtud = !socket.may_fragment();
151        let rc = EndpointRef::new(
152            socket,
153            proto::Endpoint::new(Arc::new(config), server_config.map(Arc::new), allow_mtud),
154            addr.is_ipv6(),
155            runtime.clone(),
156        );
157        let driver = EndpointDriver(rc.clone());
158        runtime.spawn(Box::pin(
159            async {
160                if let Err(e) = driver.await {
161                    tracing::error!("I/O error: {}", e);
162                }
163            }
164            .instrument(Span::current()),
165        ));
166        Ok(Self { inner: rc, runtime })
167    }

187    /// Connect to a remote endpoint
188    ///
189    /// `server_name` must be covered by the certificate presented by the server. This prevents a
190    /// connection from being intercepted by an attacker with a valid certificate for some other
191    /// server.
192    ///
193    /// May fail immediately due to configuration errors, or in the future if the connection could
194    /// not be established.
195    pub fn connect(&self, addr: SocketAddr, server_name: &str) -> Result<Connecting, ConnectError> {
196        let Some(config) = self
197            .inner
198            .0
199            .state
200            .lock()
201            .unwrap()
202            .default_client_config
203            .clone()
204        else {
205            return Err(ConnectError::NoDefaultClientConfig);
206        };
207
208        self.connect_with(config, addr, server_name)
209    }

211    /// Connect to a remote endpoint using a custom configuration.
212    ///
213    /// See [`connect()`] for details.
214    ///
215    /// [`connect()`]: Endpoint::connect
216    pub fn connect_with(
217        &self,
218        config: ClientConfig,
219        addr: SocketAddr,
220        server_name: &str,
221    ) -> Result<Connecting, ConnectError> {
222        let mut endpoint = self.inner.state.lock().unwrap();
223        if endpoint.driver_lost || endpoint.recv_state.connections.close.is_some() {
224            return Err(ConnectError::EndpointStopping);
225        }
226        if addr.is_ipv6() && !endpoint.ipv6 {
227            return Err(ConnectError::InvalidRemoteAddress(addr));
228        }
229        let addr = if endpoint.ipv6 {
230            SocketAddr::V6(ensure_ipv6(addr))
231        } else {
232            addr
233        };
234
235        let (ch, conn) = endpoint
236            .inner
237            .connect(self.runtime.now(), config, addr, server_name)?;
238
239        let sender = endpoint.socket.create_sender();
240        endpoint.stats.outgoing_handshakes += 1;
241        Ok(endpoint
242            .recv_state
243            .connections
244            .insert(ch, conn, sender, self.runtime.clone()))
245    }

255    /// Switch to a new UDP socket
256    ///
257    /// Allows the endpoint's address to be updated live, affecting all active connections. Incoming
258    /// connections and connections to servers unreachable from the new address will be lost.
259    ///
260    /// On error, the old UDP socket is retained.
261    pub fn rebind_abstract(&self, socket: Box<dyn AsyncUdpSocket>) -> io::Result<()> {
262        let addr = socket.local_addr()?;
263        let mut inner = self.inner.state.lock().unwrap();
264        inner.prev_socket = Some(mem::replace(&mut inner.socket, socket));
265        inner.ipv6 = addr.is_ipv6();
266
267        // Update connection socket references
268        for sender in inner.recv_state.connections.senders.values() {
269            // Ignoring errors from dropped connections
270            let _ = sender.send(ConnectionEvent::Rebind(inner.socket.create_sender()));
271        }
272        if let Some(driver) = inner.driver.take() {
273            // Ensure the driver can register for wake-ups from the new socket
274            driver.wake();
275        }
276
277        Ok(())
278    }

302    /// Close all of this endpoint's connections immediately and cease accepting new connections.
303    ///
304    /// See [`Connection::close()`] for details.
305    ///
306    /// [`Connection::close()`]: crate::Connection::close
307    pub fn close(&self, error_code: VarInt, reason: &[u8]) {
308        let reason = Bytes::copy_from_slice(reason);
309        let mut endpoint = self.inner.state.lock().unwrap();
310        endpoint.recv_state.connections.close = Some((error_code, reason.clone()));
311        for sender in endpoint.recv_state.connections.senders.values() {
312            // Ignoring errors from dropped connections
313            let _ = sender.send(ConnectionEvent::Close {
314                error_code,
315                reason: reason.clone(),
316            });
317        }
318        self.inner.shared.incoming.notify_waiters();
319    }
320
321    /// Wait for all connections on the endpoint to be cleanly shut down
322    ///
323    /// Waiting for this condition before exiting ensures that a good-faith effort is made to notify
324    /// peers of recent connection closes, whereas exiting immediately could force them to wait out
325    /// the idle timeout period.
326    ///
327    /// Does not proactively close existing connections or cause incoming connections to be
328    /// rejected. Consider calling [`close()`] if that is desired.
329    ///
330    /// [`close()`]: Endpoint::close
331    pub async fn wait_idle(&self) {
332        loop {
333            {
334                let endpoint = &mut *self.inner.state.lock().unwrap();
335                if endpoint.recv_state.connections.is_empty() {
336                    break;
337                }
338                // Construct future while lock is held to avoid race
339                self.inner.shared.idle.notified()
340            }
341            .await;
342        }
343    }

360/// A future that drives IO on an endpoint
361///
362/// This task functions as the switch point between the UDP socket object and the
363/// `Endpoint` responsible for routing datagrams to their owning `Connection`.
364/// In order to do so, it also facilitates the exchange of different types of events
365/// flowing between the `Endpoint` and the tasks managing `Connection`s. As such,
366/// running this task is necessary to keep the endpoint's connections running.
367///
368/// `EndpointDriver` futures terminate when all clones of the `Endpoint` have been dropped, or when
369/// an I/O error occurs.
370#[must_use = "endpoint drivers must be spawned for I/O to occur"]
371#[derive(Debug)]
372pub(crate) struct EndpointDriver(pub(crate) EndpointRef);
373
374impl Future for EndpointDriver {
375    type Output = Result<(), io::Error>;
376
377    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
378        let mut endpoint = self.0.state.lock().unwrap();
379        if endpoint.driver.is_none() {
380            endpoint.driver = Some(cx.waker().clone());
381        }
382
383        let now = endpoint.runtime.now();
384        let mut keep_going = false;
385        keep_going |= endpoint.drive_recv(cx, now)?;
386        keep_going |= endpoint.handle_events(cx, &self.0.shared);
387
388        if !endpoint.recv_state.incoming.is_empty() {
389            self.0.shared.incoming.notify_waiters();
390        }
391
392        if endpoint.ref_count == 0 && endpoint.recv_state.connections.is_empty() {
393            Poll::Ready(Ok(()))
394        } else {
395            drop(endpoint);
396            // If there is more work to do schedule the endpoint task again.
397            // `wake_by_ref()` is called outside the lock to minimize
398            // lock contention on a multithreaded runtime.
399            if keep_going {
400                cx.waker().wake_by_ref();
401            }
402            Poll::Pending
403        }
404    }
405}
406
407impl Drop for EndpointDriver {
408    fn drop(&mut self) {
409        let mut endpoint = self.0.state.lock().unwrap();
410        endpoint.driver_lost = true;
411        self.0.shared.incoming.notify_waiters();
412        // Drop all outgoing channels, signaling the termination of the endpoint to the associated
413        // connections.
414        endpoint.recv_state.connections.senders.clear();
415    }
416}

419pub(crate) struct EndpointInner {
420    pub(crate) state: Mutex<State>,
421    pub(crate) shared: Shared,
422}
423
424impl EndpointInner {
425    pub(crate) fn accept(
426        &self,
427        incoming: proto::Incoming,
428        server_config: Option<Arc<ServerConfig>>,
429    ) -> Result<Connecting, ConnectionError> {
430        let mut state = self.state.lock().unwrap();
431        let mut response_buffer = Vec::new();
432        let now = state.runtime.now();
433        match state
434            .inner
435            .accept(incoming, now, &mut response_buffer, server_config)
436        {
437            Ok((handle, conn)) => {
438                state.stats.accepted_handshakes += 1;
439                let sender = state.socket.create_sender();
440                let runtime = state.runtime.clone();
441                Ok(state
442                    .recv_state
443                    .connections
444                    .insert(handle, conn, sender, runtime))
445            }
446            Err(error) => {
447                if let Some(transmit) = error.response {
448                    respond(transmit, &response_buffer, &mut state.sender);
449                }
450                Err(error.cause)
451            }
452        }
453    }
454
455    pub(crate) fn refuse(&self, incoming: proto::Incoming) {
456        let mut state = self.state.lock().unwrap();
457        state.stats.refused_handshakes += 1;
458        let mut response_buffer = Vec::new();
459        let transmit = state.inner.refuse(incoming, &mut response_buffer);
460        respond(transmit, &response_buffer, &mut state.sender);
461    }
462
463    pub(crate) fn retry(&self, incoming: proto::Incoming) -> Result<(), proto::RetryError> {
464        let mut state = self.state.lock().unwrap();
465        let mut response_buffer = Vec::new();
466        let transmit = state.inner.retry(incoming, &mut response_buffer)?;
467        respond(transmit, &response_buffer, &mut state.sender);
468        Ok(())
469    }
470
471    pub(crate) fn ignore(&self, incoming: proto::Incoming) {
472        let mut state = self.state.lock().unwrap();
473        state.stats.ignored_handshakes += 1;
474        state.inner.ignore(incoming);
475    }
476}

479pub(crate) struct State {
480    socket: Box<dyn AsyncUdpSocket>,
481    sender: Pin<Box<dyn UdpSender>>,
482    /// During an active migration, abandoned_socket receives traffic
483    /// until the first packet arrives on the new socket.
484    prev_socket: Option<Box<dyn AsyncUdpSocket>>,
485    inner: proto::Endpoint,
486    recv_state: RecvState,
487    driver: Option<Waker>,
488    ipv6: bool,
489    events: mpsc::UnboundedReceiver<(ConnectionHandle, EndpointEvent)>,
490    /// Number of live handles that can be used to initiate or handle I/O; excludes the driver
491    ref_count: usize,
492    driver_lost: bool,
493    runtime: Arc<dyn Runtime>,
494    stats: EndpointStats,
495    default_client_config: Option<ClientConfig>,
496}

498#[derive(Debug)]
499pub(crate) struct Shared {
500    incoming: Notify,
501    idle: Notify,
502}
503
504impl State {
505    fn drive_recv(&mut self, cx: &mut Context<'_>, now: Instant) -> Result<bool, io::Error> {
506        let get_time = || self.runtime.now();
507        self.recv_state.recv_limiter.start_cycle(get_time);
508        if let Some(socket) = &mut self.prev_socket {
509            // We don't care about the `PollProgress` from old sockets.
510            let poll_res = self.recv_state.poll_socket(
511                cx,
512                &mut self.inner,
513                &mut **socket,
514                &mut self.sender,
515                &*self.runtime,
516                now,
517            );
518            if poll_res.is_err() {
519                self.prev_socket = None;
520            }
521        };
522        let poll_res = self.recv_state.poll_socket(
523            cx,
524            &mut self.inner,
525            &mut *self.socket,
526            &mut self.sender,
527            &*self.runtime,
528            now,
529        );
530        self.recv_state.recv_limiter.finish_cycle(get_time);
531        let poll_res = poll_res?;
532        if poll_res.received_connection_packet {
533            // Traffic has arrived on self.socket, therefore there is no need for the abandoned
534            // one anymore. TODO: Account for multiple outgoing connections.
535            self.prev_socket = None;
536        }
537        Ok(poll_res.keep_going)
538    }
539
540    fn handle_events(&mut self, cx: &mut Context<'_>, shared: &Shared) -> bool {
541        for _ in 0..IO_LOOP_BOUND {
542            let (ch, event) = match self.events.poll_recv(cx) {
543                Poll::Ready(Some(x)) => x,
544                Poll::Ready(None) => unreachable!("EndpointInner owns one sender"),
545                Poll::Pending => {
546                    return false;
547                }
548            };
549
550            if event.is_drained() {
551                self.recv_state.connections.senders.remove(&ch);
552                if self.recv_state.connections.is_empty() {
553                    shared.idle.notify_waiters();
554                }
555            }
556            let Some(event) = self.inner.handle_event(ch, event) else {
557                continue;
558            };
559            // Ignoring errors from dropped connections that haven't yet been cleaned up
560            let _ = self
561                .recv_state
562                .connections
563                .senders
564                .get_mut(&ch)
565                .unwrap()
566                .send(ConnectionEvent::Proto(event));
567        }
568
569        true
570    }
571}
572
573impl Drop for State {
574    fn drop(&mut self) {
575        for incoming in self.recv_state.incoming.drain(..) {
576            self.inner.ignore(incoming);
577        }
578    }
579}

581fn respond(
582    transmit: proto::Transmit,
583    response_buffer: &[u8],
584    sender: &mut Pin<Box<dyn UdpSender>>,
585) {
586    // Send if there's kernel buffer space; otherwise, drop it
587    //
588    // As an endpoint-generated packet, we know this is an
589    // immediate, stateless response to an unconnected peer,
590    // one of:
591    //
592    // - A version negotiation response due to an unknown version
593    // - A `CLOSE` due to a malformed or unwanted connection attempt
594    // - A stateless reset due to an unrecognized connection
595    // - A `Retry` packet due to a connection attempt when
596    //   `use_retry` is set
597    //
598    // In each case, a well-behaved peer can be trusted to retry a
599    // few times, which is guaranteed to produce the same response
600    // from us. Repeated failures might at worst cause a peer's new
601    // connection attempt to time out, which is acceptable if we're
602    // under such heavy load that there's never room for this code
603    // to transmit. This is morally equivalent to the packet getting
604    // lost due to congestion further along the link, which
605    // similarly relies on peer retries for recovery.
606
607    // Copied from rust 1.85's std::task::Waker::noop() implementation for backwards compatibility
608    const NOOP: RawWaker = {
609        const VTABLE: RawWakerVTable = RawWakerVTable::new(
610            // Cloning just returns a new no-op raw waker
611            |_| NOOP,
612            // `wake` does nothing
613            |_| {},
614            // `wake_by_ref` does nothing
615            |_| {},
616            // Dropping does nothing as we don't allocate anything
617            |_| {},
618        );
619        RawWaker::new(std::ptr::null(), &VTABLE)
620    };
621    // SAFETY: Copied from rust stdlib, the NOOP waker is thread-safe and doesn't violate the RawWakerVTable contract,
622    // it doesn't access the data pointer at all.
623    let waker = unsafe { Waker::from_raw(NOOP) };
624    let mut cx = Context::from_waker(&waker);
625    _ = sender.as_mut().poll_send(
626        &udp_transmit(&transmit, &response_buffer[..transmit.size]),
627        &mut cx,
628    );
629}

640#[derive(Debug)]
641struct ConnectionSet {
642    /// Senders for communicating with the endpoint's connections
643    senders: FxHashMap<ConnectionHandle, mpsc::UnboundedSender<ConnectionEvent>>,
644    /// Stored to give out clones to new ConnectionInners
645    sender: mpsc::UnboundedSender<(ConnectionHandle, EndpointEvent)>,
646    /// Set if the endpoint has been manually closed
647    close: Option<(VarInt, Bytes)>,
648}
649
650impl ConnectionSet {
651    fn insert(
652        &mut self,
653        handle: ConnectionHandle,
654        conn: proto::Connection,
655        sender: Pin<Box<dyn UdpSender>>,
656        runtime: Arc<dyn Runtime>,
657    ) -> Connecting {
658        let (send, recv) = mpsc::unbounded_channel();
659        if let Some((error_code, ref reason)) = self.close {
660            send.send(ConnectionEvent::Close {
661                error_code,
662                reason: reason.clone(),
663            })
664            .unwrap();
665        }
666        self.senders.insert(handle, send);
667        Connecting::new(handle, conn, self.sender.clone(), recv, sender, runtime)
668    }
669
670    fn is_empty(&self) -> bool {
671        self.senders.is_empty()
672    }
673}

678        SocketAddr::V4(x) => SocketAddrV6::new(x.ip().to_ipv6_mapped(), x.port(), 0, 0),
679    }

682pin_project! {
683    /// Future produced by [`Endpoint::accept`]
684    pub struct Accept<'a> {
685        endpoint: &'a Endpoint,
686        #[pin]
687        notify: Notified<'a>,
688    }
689}
690
691impl Future for Accept<'_> {
692    type Output = Option<Incoming>;
693    fn poll(self: Pin<&mut Self>, ctx: &mut Context<'_>) -> Poll<Self::Output> {
694        let mut this = self.project();
695        let mut endpoint = this.endpoint.inner.state.lock().unwrap();
696        if endpoint.driver_lost {
697            return Poll::Ready(None);
698        }
699        if let Some(incoming) = endpoint.recv_state.incoming.pop_front() {
700            // Release the mutex lock on endpoint so cloning it doesn't deadlock
701            drop(endpoint);
702            let incoming = Incoming::new(incoming, this.endpoint.inner.clone());
703            return Poll::Ready(Some(incoming));
704        }
705        if endpoint.recv_state.connections.close.is_some() {
706            return Poll::Ready(None);
707        }
708        loop {
709            match this.notify.as_mut().poll(ctx) {
710                // `state` lock ensures we didn't race with readiness
711                Poll::Pending => return Poll::Pending,
712                // Spurious wakeup, get a new future
713                Poll::Ready(()) => this
714                    .notify
715                    .set(this.endpoint.inner.shared.incoming.notified()),
716            }
717        }
718    }
719}

754        }))
755    }

758impl Clone for EndpointRef {
759    fn clone(&self) -> Self {
760        self.0.state.lock().unwrap().ref_count += 1;
761        Self(self.0.clone())
762    }
763}
764
765impl Drop for EndpointRef {
766    fn drop(&mut self) {
767        let endpoint = &mut *self.0.state.lock().unwrap();
768        if let Some(x) = endpoint.ref_count.checked_sub(1) {
769            endpoint.ref_count = x;
770            if x == 0 {
771                // If the driver is about to be on its own, ensure it can shut down if the last
772                // connection is gone.
773                if let Some(task) = endpoint.driver.take() {
774                    task.wake();
775                }
776            }
777        }
778    }
779}

784        &self.0
785    }

788/// State directly involved in handling incoming packets
789struct RecvState {
790    incoming: VecDeque<proto::Incoming>,
791    connections: ConnectionSet,
792    recv_limiter: WorkLimiter,
793}
794
795impl RecvState {
796    fn new(
797        sender: mpsc::UnboundedSender<(ConnectionHandle, EndpointEvent)>,
798    ) -> Self {
799        Self {
800            connections: ConnectionSet {
801                senders: FxHashMap::default(),
802                sender,
803                close: None,
804            },
805            incoming: VecDeque::new(),
806            recv_limiter: WorkLimiter::new(RECV_TIME_BOUND),
807        }
808    }
809
810    fn poll_socket(
811        &mut self,
812        cx: &mut Context<'_>,
813        endpoint: &mut proto::Endpoint,
814        socket: &mut dyn AsyncUdpSocket,
815        sender: &mut Pin<Box<dyn UdpSender>>,
816        runtime: &dyn Runtime,
817        now: Instant,
818    ) -> Result<PollProgress, io::Error> {
819        let mut received_connection_packet = false;
820        loop {
821            match socket.poll_recv_datagrams(cx) {
822                Poll::Ready(Ok(datagrams)) => {
823                    self.recv_limiter.record_work(datagrams.len());
824                    for datagram in datagrams {
825                        let mut response_buffer = Vec::new();
826                        let addresses = FourTuple {
827                            remote: datagram.remote,
828                            local_ip: datagram.local_ip,
829                        };
830                        match endpoint.handle(
831                            now,
832                            addresses,
833                            datagram.ecn.map(proto_ecn),
834                            datagram.data,
835                            &mut response_buffer,
836                        ) {
837                            Some(DatagramEvent::NewConnection(incoming)) => {
838                                if self.connections.close.is_none() {
839                                    self.incoming.push_back(incoming);
840                                } else {
841                                    let transmit =
842                                        endpoint.refuse(incoming, &mut response_buffer);
843                                    respond(transmit, &response_buffer, sender);
844                                }
845                            }
846                            Some(DatagramEvent::ConnectionEvent(handle, event)) => {
847                                // Ignoring errors from dropped connections that haven't yet been cleaned up
848                                received_connection_packet = true;
849                                let _ = self
850                                    .connections
851                                    .senders
852                                    .get_mut(&handle)
853                                    .unwrap()
854                                    .send(ConnectionEvent::Proto(event));
855                            }
856                            Some(DatagramEvent::Response(transmit)) => {
857                                respond(transmit, &response_buffer, sender);
858                            }
859                            None => {}
860                        }
861                    }
862                }
863                Poll::Pending => {
864                    return Ok(PollProgress {
865                        received_connection_packet,
866                        keep_going: false,
867                    });
868                }
869                // Ignore ECONNRESET as it's undefined in QUIC and may be injected by an
870                // attacker
871                Poll::Ready(Err(ref e)) if e.kind() == io::ErrorKind::ConnectionReset => {
872                    continue;
873                }
874                Poll::Ready(Err(e)) => {
875                    return Err(e);
876                }
877            }
878            if !self.recv_limiter.allow_work(|| runtime.now()) {
879                return Ok(PollProgress {
880                    received_connection_packet,
881                    keep_going: true,
882                });
883            }
884        }
885    }
886}
887
888impl fmt::Debug for RecvState {
889    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
890        f.debug_struct("RecvState")
891            .field("incoming", &self.incoming)
892            .field("connections", &self.connections)
893            .field("recv_limiter", &self.recv_limiter)
894            .finish_non_exhaustive()
895    }
896}
897
898#[derive(Default)]
899struct PollProgress {
900    /// Whether a datagram was routed to an existing connection
901    received_connection_packet: bool,
902    /// Whether datagram handling was interrupted early by the work limiter for fairness
903    keep_going: bool,
904}
