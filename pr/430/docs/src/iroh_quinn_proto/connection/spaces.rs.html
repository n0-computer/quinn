86    /// Queue data for a tail loss probe (or anti-amplification deadlock prevention) packet
87    ///
88    /// Does nothing if no tail loss probe needs to be sent.
89    ///
90    /// Probes are sent similarly to normal packets when an expected ACK has not arrived. We never
91    /// deem a packet lost until we receive an ACK that should have included it, but if a trailing
92    /// run of packets (or their ACKs) are lost, this might not happen in a timely fashion. We send
93    /// probe packets to force an ACK, and exempt them from congestion control to prevent a deadlock
94    /// when the congestion window is filled with lost tail packets.
95    ///
96    /// We prefer to send new data, to make the most efficient use of bandwidth. If there's no data
97    /// waiting to be sent, then we retransmit in-flight data to reduce odds of loss. If there's no
98    /// in-flight data either, we're probably a client guarding against a handshake
99    /// anti-amplification deadlock and we just make something up.
100    pub(super) fn maybe_queue_probe(
101        &mut self,
102        path_id: PathId,
103        request_immediate_ack: bool,
104        streams: &StreamsState,
105    ) {
106        if self.for_path(path_id).loss_probes == 0 {
107            return;
108        }
109
110        if request_immediate_ack {
111            // The probe should be ACKed without delay (should only be used in the Data space and
112            // when the peer supports the acknowledgement frequency extension)
113            self.for_path(path_id).immediate_ack_pending = true;
114        }
115
116        // Retransmit the data of the oldest in-flight packet
117        if !self.pending.is_empty(streams) {
118            // There's real data to send here, no need to make something up
119            return;
120        }
121
122        // We use retransmits from any path.
123        for packet in self
124            .number_spaces
125            .values_mut()
126            .flat_map(|s| s.sent_packets.values_mut())
127        {
128            if !packet.retransmits.is_empty(streams) {
129                // Remove retransmitted data from the old packet so we don't end up retransmitting
130                // it *again* even if the copy we're sending now gets acknowledged.
131                self.pending |= mem::take(&mut packet.retransmits);
132                return;
133            }
134        }
135
136        // Nothing new to send and nothing to retransmit, so fall back on a ping. This should only
137        // happen in rare cases during the handshake when the server becomes blocked by
138        // anti-amplification.
139        if !self.for_path(path_id).immediate_ack_pending {
140            self.for_path(path_id).ping_pending = true;
141        }
142    }
143
144    /// Whether there is anything to send in this space
145    ///
146    /// For the data space [`Connection::can_send_1rtt`] also needs to be consulted.
147    ///
148    /// [`Connection::can_send_1rtt`]: super::Connection::can_send_1rtt
149    pub(super) fn can_send(&self, path_id: PathId, streams: &StreamsState) -> SendableFrames {
150        let acks = self
151            .number_spaces
152            .values()
153            .any(|pns| pns.pending_acks.can_send());
154        let path_exclusive = self
155            .number_spaces
156            .get(&path_id)
157            .is_some_and(|s| s.ping_pending || s.immediate_ack_pending);
158        let other = !self.pending.is_empty(streams) || path_exclusive;
159        SendableFrames {
160            acks,
161            other,
162            close: false,
163            path_exclusive,
164        }
165    }
166}

181/// The per-path packet number space to support multipath.
182///
183/// This contains the data specific to a per-path packet number space.  You should access
184/// this via [`PacketSpace::for_path`].
185pub(super) struct PacketNumberSpace {
186    /// Highest received packet number, if any
187    pub(super) rx_packet: Option<u64>,
188    /// The packet number of the next packet that will be sent, if any. In the Data space, the
189    /// packet number stored here is sometimes skipped by [`PacketNumberFilter`] logic.
190    pub(super) next_packet_number: u64,
191    /// The largest packet number the remote peer acknowledged in an ACK frame.
192    pub(super) largest_acked_packet: Option<u64>,
193    pub(super) largest_acked_packet_sent: Instant,
194    /// The highest-numbered ACK-eliciting packet we've sent
195    pub(super) largest_ack_eliciting_sent: u64,
196    /// Number of packets in `sent_packets` with numbers above `largest_ack_eliciting_sent`
197    pub(super) unacked_non_ack_eliciting_tail: u64,

200    pub(super) sent_packets: SortedIndexBuffer<SentPacket>,

202    // Older packets are regularly removed in `Connection::drain_lost_packets`.
203    pub(super) lost_packets: SortedIndexBuffer<LostPacket>,

211    /// ACK already processed in another space.

213    /// A PING frame needs to be sent on this path
214    pub(super) ping_pending: bool,

216    pub(super) immediate_ack_pending: bool,

219    /// Packet numbers to acknowledge

224    //

226    pub(super) time_of_last_ack_eliciting_packet: Option<Instant>,
227    /// Earliest time when we might declare a packet lost.
228    ///
229    /// The time at which the earliest sent packet in this space will be considered lost

296    /// Get the next outgoing packet number in this space
297    ///
298    /// In the Data space, the connection's [`PacketNumberFilter`] must be used rather than calling
299    /// this directly.
300    pub(super) fn get_tx_number(&mut self, rng: &mut (impl Rng + ?Sized)) -> u64 {
301        // TODO: Handle packet number overflow gracefully
302        assert!(self.next_packet_number < 2u64.pow(62));
303        let mut pn = self.next_packet_number;
304        self.next_packet_number += 1;
305
306        // Skip this number if the filter says so, only enabled in the data space
307        if let Some(ref mut filter) = self.pn_filter
308            && filter.skip_pn(pn, rng)
309        {
310            pn = self.next_packet_number;
311            self.next_packet_number += 1;
312        }
313        pn
314    }

316    pub(super) fn peek_tx_number(&mut self) -> u64 {
317        let pn = self.next_packet_number;
318        if let Some(ref filter) = self.pn_filter
319            && pn == filter.next_skipped_packet_number
320        {
321            return pn + 1;
322        }
323        pn
324    }

338    /// Verifies sanity of an ECN block and returns whether congestion was encountered.
339    pub(super) fn detect_ecn(
340        &mut self,
341        newly_acked: u64,
342        ecn: frame::EcnCounts,
343    ) -> Result<bool, &'static str> {
344        let ect0_increase = ecn
345            .ect0
346            .checked_sub(self.ecn_feedback.ect0)
347            .ok_or("peer ECT(0) count regression")?;
348        let ect1_increase = ecn
349            .ect1
350            .checked_sub(self.ecn_feedback.ect1)
351            .ok_or("peer ECT(1) count regression")?;
352        let ce_increase = ecn
353            .ce
354            .checked_sub(self.ecn_feedback.ce)
355            .ok_or("peer CE count regression")?;
356        let total_increase = ect0_increase + ect1_increase + ce_increase;
357        if total_increase < newly_acked {
358            return Err("ECN bleaching");
359        }
360        if (ect0_increase + ce_increase) < newly_acked || ect1_increase != 0 {
361            return Err("ECN corruption");
362        }
363        // If total_increase > newly_acked (which happens when ACKs are lost), this is required by
364        // the draft so that long-term drift does not occur. If =, then the only question is whether
365        // to count CE packets as CE or ECT0. Recording them as CE is more consistent and keeps the
366        // congestion check obvious.
367        self.ecn_feedback = ecn;
368        Ok(ce_increase != 0)
369    }

371    /// Stop tracking sent packet `number`, and return what we knew about it
372    pub(super) fn take(&mut self, number: u64) -> Option<SentPacket> {
373        let packet = self.sent_packets.remove(number)?;
374        if !packet.ack_eliciting && number > self.largest_ack_eliciting_sent {
375            self.unacked_non_ack_eliciting_tail =
376                self.unacked_non_ack_eliciting_tail.checked_sub(1).unwrap();
377        }
378        Some(packet)
379    }
380
381    /// May return a packet that should be forgotten
382    pub(super) fn sent(&mut self, number: u64, packet: SentPacket) -> Option<SentPacket> {
383        // Retain state for at most this many non-ACK-eliciting packets sent after the most recently
384        // sent ACK-eliciting packet. We're never guaranteed to receive an ACK for those, and we
385        // can't judge them as lost without an ACK, so to limit memory in applications which receive
386        // packets but don't send ACK-eliciting data for long periods use we must eventually start
387        // forgetting about them, although it might also be reasonable to just kill the connection
388        // due to weird peer behavior.
389        const MAX_UNACKED_NON_ACK_ELICTING_TAIL: u64 = 1_000;
390
391        let mut forgotten = None;
392        if packet.ack_eliciting {
393            self.unacked_non_ack_eliciting_tail = 0;
394            self.largest_ack_eliciting_sent = number;
395        } else if self.unacked_non_ack_eliciting_tail > MAX_UNACKED_NON_ACK_ELICTING_TAIL {
396            let oldest_after_ack_eliciting = self
397                .sent_packets
398                .keys_range((
399                    Bound::Excluded(self.largest_ack_eliciting_sent),
400                    Bound::Unbounded,
401                ))
402                .next()
403                .unwrap();
404            // Per https://www.rfc-editor.org/rfc/rfc9000.html#name-frames-and-frame-types,
405            // non-ACK-eliciting packets must only contain PADDING, ACK, and CONNECTION_CLOSE
406            // frames, which require no special handling on ACK or loss beyond removal from
407            // in-flight counters if padded.
408            let packet = self
409                .sent_packets
410                .remove(oldest_after_ack_eliciting)
411                .unwrap();
412            debug_assert!(!packet.ack_eliciting);
413            forgotten = Some(packet);
414        } else {
415            self.unacked_non_ack_eliciting_tail += 1;
416        }
417
418        self.sent_packets.insert(number, packet);
419        forgotten
420    }

422    /// Whether any congestion-controlled packets in this space are not yet acknowledged or lost
423    pub(super) fn has_in_flight(&self) -> bool {
424        // The number of non-congestion-controlled (i.e. size == 0) packets in flight at a time
425        // should be small, since otherwise congestion control wouldn't be effective. Therefore,
426        // this shouldn't need to visit many packets before finishing one way or another.
427        self.sent_packets.values().any(|x| x.size != 0)
428    }

463/// Retransmittable data queue
464#[allow(unreachable_pub)] // fuzzing only

468    pub(super) max_stream_id: [bool; 2],
469    pub(super) reset_stream: Vec<(StreamId, VarInt)>,
470    pub(super) stop_sending: Vec<frame::StopSending>,
471    pub(super) max_stream_data: FxHashSet<StreamId>,
472    pub(super) crypto: VecDeque<frame::Crypto>,
473    pub(super) new_cids: Vec<IssuedCid>,
474    pub(super) retire_cids: Vec<(PathId, u64)>,
475    pub(super) ack_frequency: bool,
476    pub(super) handshake_done: bool,
477    pub(super) observed_addr: bool,
478    /// Whether we should inform the peer we will allow higher [`PathId`]s.
479    pub(super) max_path_id: bool,
480    /// Whether we should inform the peer that their max [`PathId`] is blocking our attempt to open
481    /// new paths.
482    // TODO(@divma): we need logic to prevent sending this more than once after being ack-d once
483    pub(super) paths_blocked: bool,
484    /// For each enqueued NEW_TOKEN frame, a copy of the path's remote address
485    ///
486    /// There are 2 reasons this is unusual:
487    ///
488    /// - If the path changes, NEW_TOKEN frames bound for the old path are not retransmitted on the
489    ///   new path. That is why this field stores the remote address: so that ones for old paths
490    ///   can be filtered out.
491    /// - If a token is lost, a new randomly generated token is re-transmitted, rather than the
492    ///   original. This is so that if both transmissions are received, the client won't risk
493    ///   sending the same token twice. That is why this field does _not_ store any actual token.
494    ///
495    /// It is true that a QUIC endpoint will only want to effectively have NEW_TOKEN frames
496    /// enqueued for its current path at a given point in time. Based on that, we could conceivably
497    /// change this from a vector to an `Option<(FourTuple, usize)>` or just a `usize` or
498    /// something. However, due to the architecture of Quinn, it is considerably simpler to not do
499    /// that; consider what such a change would mean for implementing `BitOrAssign` on Self.
500    pub(super) new_tokens: Vec<FourTuple>,
501    /// Paths which need to be abandoned
502    pub(super) path_abandon: BTreeMap<PathId, TransportErrorCode>,
503    /// If a [`frame::PathStatusAvailable`] and [`frame::PathStatusBackup`] need to be sent for a path
504    pub(super) path_status: BTreeSet<PathId>,
505    /// If a PATH_CIDS_BLOCKED frame needs to be sent for a path
506    pub(super) path_cids_blocked: BTreeSet<PathId>,
507
508    // Nat traversal data
509    /// Addresses to report in `ADD_ADDRESS` frames
510    pub(super) add_address: BTreeSet<AddAddress>,
511    /// Address IDs to remove in `REMOVE_ADDRESS` frames
512    pub(super) remove_address: BTreeSet<RemoveAddress>,
513    /// Round and local addresses to advertise in `REACH_OUT` frames
514    pub(super) reach_out: Option<(VarInt, FxHashSet<(IpAddr, u16)>)>,
515}

517impl Retransmits {
518    pub(super) fn is_empty(&self, streams: &StreamsState) -> bool {
519        !self.max_data
520            && !self.max_stream_id.into_iter().any(|x| x)
521            && self.reset_stream.is_empty()
522            && self.stop_sending.is_empty()
523            && self
524                .max_stream_data
525                .iter()
526                .all(|&id| !streams.can_send_flow_control(id))
527            && self.crypto.is_empty()
528            && self.new_cids.is_empty()
529            && self.retire_cids.is_empty()
530            && !self.ack_frequency
531            && !self.handshake_done
532            && !self.observed_addr
533            && self.new_tokens.is_empty()
534            && self.path_abandon.is_empty()
535            && self.path_status.is_empty()
536            && !self.max_path_id
537            && !self.paths_blocked
538            && self.add_address.is_empty()
539            && self.remove_address.is_empty()
540            && self.reach_out.is_none()
541    }

544impl ::std::ops::BitOrAssign for Retransmits {
545    fn bitor_assign(&mut self, mut rhs: Self) {
546        // We reduce in-stream head-of-line blocking by queueing retransmits before other data for
547        // STREAM and CRYPTO frames.
548        self.max_data |= rhs.max_data;
549        for dir in Dir::iter() {
550            self.max_stream_id[dir as usize] |= rhs.max_stream_id[dir as usize];
551        }
552        self.reset_stream.extend_from_slice(&rhs.reset_stream);
553        self.stop_sending.extend_from_slice(&rhs.stop_sending);
554        self.max_stream_data.extend(&rhs.max_stream_data);
555        for crypto in rhs.crypto.into_iter().rev() {
556            self.crypto.push_front(crypto);
557        }
558        self.new_cids.extend(&rhs.new_cids);
559        self.retire_cids.extend(rhs.retire_cids);
560        self.ack_frequency |= rhs.ack_frequency;
561        self.handshake_done |= rhs.handshake_done;
562        self.observed_addr |= rhs.observed_addr;
563        self.new_tokens.extend_from_slice(&rhs.new_tokens);
564        self.path_abandon.append(&mut rhs.path_abandon);
565        self.max_path_id |= rhs.max_path_id;
566        self.paths_blocked |= rhs.paths_blocked;
567        self.add_address.extend(rhs.add_address.iter().copied());
568        self.remove_address
569            .extend(rhs.remove_address.iter().copied());
570        if let Some((rhs_round, rhs_addrs)) = rhs.reach_out {
571            match self.reach_out.as_mut() {
572                // Use RHS if there is no recorded round.
573                None => self.reach_out = Some((rhs_round, rhs_addrs)),
574                // Use RHS if newer.
575                Some((lhs_round, lhs_addrs)) if rhs_round > *lhs_round => {
576                    self.reach_out = Some((rhs_round, rhs_addrs));
577                }
578                // If both rounds are the same, merge them.
579                Some((lhs_round, lhs_addrs)) if rhs_round == *lhs_round => {
580                    lhs_addrs.extend(rhs_addrs);
581                }
582                // LHS round is newer, ignore RHS
583                Some(_) => {}
584            }
585        }
586    }
587}

646/// RFC4303-style sliding window packet number deduplicator.
647///
648/// A contiguous bitfield, where each bit corresponds to a packet number and the rightmost bit is
649/// always set. A set bit represents a packet that has been successfully authenticated. Bits left of
650/// the window are assumed to be set.
651///
652/// ```text
653/// ...xxxxxxxxx 1 0
654///     ^        ^ ^
655/// window highest next
656/// ```
657#[derive(Debug, Default)]
658pub(super) struct Dedup {
659    window: Window,
660    /// Lowest packet number higher than all yet authenticated.
661    next: u64,
662}
663
664/// Inner bitfield type.
665///
666/// Because QUIC never reuses packet numbers, this only needs to be large enough to deal with
667/// packets that are reordered but still delivered in a timely manner.
668type Window = u128;
669
670/// Number of packets tracked by `Dedup`.
671const WINDOW_SIZE: u64 = 1 + mem::size_of::<Window>() as u64 * 8;
672
673impl Dedup {
674    /// Construct an empty window positioned at the start.
675    #[cfg(test)]
676    pub(super) fn new() -> Self {
677        Self { window: 0, next: 0 }
678    }
679
680    /// Highest packet number authenticated.
681    fn highest(&self) -> u64 {
682        self.next - 1
683    }
684
685    /// Record a newly authenticated packet number.
686    ///
687    /// Returns whether the packet might be a duplicate.
688    pub(super) fn insert(&mut self, packet: u64) -> bool {
689        if let Some(diff) = packet.checked_sub(self.next) {
690            // Right of window
691            self.window = ((self.window << 1) | 1)
692                .checked_shl(cmp::min(diff, u64::from(u32::MAX)) as u32)
693                .unwrap_or(0);
694            self.next = packet + 1;
695            false
696        } else if self.highest() - packet < WINDOW_SIZE {
697            // Within window
698            if let Some(bit) = (self.highest() - packet).checked_sub(1) {
699                // < highest
700                let mask = 1 << bit;
701                let duplicate = self.window & mask != 0;
702                self.window |= mask;
703                duplicate
704            } else {
705                // == highest
706                true
707            }
708        } else {
709            // Left of window
710            true
711        }
712    }
713
714    /// Returns the packet number of the smallest packet missing between the provided interval
715    ///
716    /// If there are no missing packets, returns `None`
717    fn smallest_missing_in_interval(&self, lower_bound: u64, upper_bound: u64) -> Option<u64> {
718        debug_assert!(lower_bound <= upper_bound);
719        debug_assert!(upper_bound <= self.highest());
720        const BITFIELD_SIZE: u64 = (mem::size_of::<Window>() * 8) as u64;
721
722        // Since we already know the packets at the boundaries have been received, we only need to
723        // check those in between them (this removes the necessity of extra logic to deal with the
724        // highest packet, which is stored outside the bitfield)
725        let lower_bound = lower_bound + 1;
726        let upper_bound = upper_bound.saturating_sub(1);
727
728        // Note: the offsets are counted from the right
729        // The highest packet is not included in the bitfield, so we subtract 1 to account for that
730        let start_offset = (self.highest() - upper_bound).max(1) - 1;
731        if start_offset >= BITFIELD_SIZE {
732            // The start offset is outside of the window. All packets outside of the window are
733            // considered to be received.
734            return None;
735        }
736
737        let end_offset_exclusive = self.highest().saturating_sub(lower_bound);
738
739        // The range is clamped at the edge of the window, because any earlier packets are
740        // considered to be received
741        let range_len = end_offset_exclusive
742            .saturating_sub(start_offset)
743            .min(BITFIELD_SIZE);
744        if range_len == 0 {
745            return None;
746        }
747
748        // Ensure the shift is within bounds (we already know start_offset < BITFIELD_SIZE,
749        // because of the early return)
750        let mask = if range_len == BITFIELD_SIZE {
751            u128::MAX
752        } else {
753            ((1u128 << range_len) - 1) << start_offset
754        };
755        let gaps = !self.window & mask;
756
757        let smallest_missing_offset = 128 - gaps.leading_zeros() as u64;
758        let smallest_missing_packet = self.highest() - smallest_missing_offset;
759
760        if smallest_missing_packet <= upper_bound {
761            Some(smallest_missing_packet)
762        } else {
763            None
764        }
765    }
766
767    /// Returns true if there are any missing packets between the provided interval
768    ///
769    /// The provided packet numbers must have been received before calling this function
770    fn missing_in_interval(&self, lower_bound: u64, upper_bound: u64) -> bool {
771        self.smallest_missing_in_interval(lower_bound, upper_bound)
772            .is_some()
773    }
774}

819#[derive(Debug)]
820pub(super) struct PendingAcks {
821    /// Whether we should send an ACK immediately, even if that means sending an ACK-only packet
822    ///
823    /// When `immediate_ack_required` is false, the normal behavior is to send ACK frames only when
824    /// there is other data to send, or when the `MaxAckDelay` timer expires.
825    immediate_ack_required: bool,
826    /// The number of ack-eliciting packets received since the last ACK frame was sent
827    ///
828    /// Once the count _exceeds_ `ack_eliciting_threshold`, an immediate ACK is required
829    ack_eliciting_since_last_ack_sent: u64,
830    non_ack_eliciting_since_last_ack_sent: u64,
831    ack_eliciting_threshold: u64,
832    /// The reordering threshold, controlling how we respond to out-of-order ack-eliciting packets
833    ///
834    /// Different values enable different behavior:
835    ///
836    /// * `0`: no special action is taken
837    /// * `1`: an ACK is immediately sent if it is out-of-order according to RFC 9000
838    /// * `>1`: an ACK is immediately sent if it is out-of-order according to the ACK frequency draft
839    reordering_threshold: u64,
840    /// The earliest ack-eliciting packet since the last ACK was sent, used to calculate the moment
841    /// upon which `max_ack_delay` elapses
842    earliest_ack_eliciting_since_last_ack_sent: Option<Instant>,
843    /// Packet number ranges for which to still send acknowledgements.
844    ///
845    /// These are packet number ranges of ack-eliciting packets the peer has sent and which
846    /// need to be acknowledged.  Packet numbers are only removed from here once the peer has
847    /// acknowledged the ACKs for them.
848    ranges: ArrayRangeSet,
849    /// The largest packet number received and the time it was received
850    ///
851    /// Used to calculate ACK delay in [`PendingAcks::ack_delay`].
852    largest_packet: Option<(u64, Instant)>,
853    /// The ack-eliciting packet we have received with the largest packet number
854    largest_ack_eliciting_packet: Option<u64>,
855    /// The largest acknowledged packet number sent in an ACK frame
856    largest_acked: Option<u64>,
857}
858
859impl PendingAcks {
860    fn new() -> Self {
861        Self {
862            immediate_ack_required: false,
863            ack_eliciting_since_last_ack_sent: 0,
864            non_ack_eliciting_since_last_ack_sent: 0,
865            ack_eliciting_threshold: 1,
866            reordering_threshold: 1,
867            earliest_ack_eliciting_since_last_ack_sent: None,
868            ranges: Default::default(),
869            largest_packet: Default::default(),
870            largest_ack_eliciting_packet: Default::default(),
871            largest_acked: Default::default(),
872        }
873    }
874
875    pub(super) fn set_ack_frequency_params(&mut self, frame: &frame::AckFrequency) {
876        self.ack_eliciting_threshold = frame.ack_eliciting_threshold.into_inner();
877        self.reordering_threshold = frame.reordering_threshold.into_inner();
878    }
879
880    pub(super) fn set_immediate_ack_required(&mut self) {
881        self.immediate_ack_required = true;
882    }
883
884    pub(super) fn on_max_ack_delay_timeout(&mut self) {
885        self.immediate_ack_required = self.ack_eliciting_since_last_ack_sent > 0;
886    }
887
888    pub(super) fn max_ack_delay_timeout(&self, max_ack_delay: Duration) -> Option<Instant> {
889        self.earliest_ack_eliciting_since_last_ack_sent
890            .map(|earliest_unacked| earliest_unacked + max_ack_delay)
891    }
892
893    /// Whether any ACK frames SHOULD be sent
894    ///
895    /// This is used in the top-level [`Connection::space_can_send`], so determines if a
896    /// packet will be built. It is often possible to construct new ACK ranges to send
897    /// before this returns `true`. This results in more ACK frames being sent, and
898    /// processing those at the receiver costs CPU for very little improvements.
899    ///
900    /// [`Connection::space_can_send`]: super::Connection::space_can_send
901    pub(super) fn can_send(&self) -> bool {
902        self.immediate_ack_required && !self.ranges.is_empty()
903    }
904
905    /// Returns the delay since the packet with the largest packet number was received
906    pub(super) fn ack_delay(&self, now: Instant) -> Duration {
907        self.largest_packet
908            .map_or_else(Duration::default, |(_, received)| now - received)
909    }
910
911    /// Handle receipt of a new packet
912    ///
913    /// Returns true if the max ack delay timer should be armed
914    pub(super) fn packet_received(
915        &mut self,
916        now: Instant,
917        packet_number: u64,
918        ack_eliciting: bool,
919        dedup: &Dedup,
920    ) -> bool {
921        if !ack_eliciting {
922            self.non_ack_eliciting_since_last_ack_sent += 1;
923            return false;
924        }
925
926        let prev_largest_ack_eliciting = self.largest_ack_eliciting_packet.unwrap_or(0);
927
928        // Track largest ack-eliciting packet
929        self.largest_ack_eliciting_packet = self
930            .largest_ack_eliciting_packet
931            .map(|pn| pn.max(packet_number))
932            .or(Some(packet_number));
933
934        // Handle ack_eliciting_threshold
935        self.ack_eliciting_since_last_ack_sent += 1;
936        self.immediate_ack_required |=
937            self.ack_eliciting_since_last_ack_sent > self.ack_eliciting_threshold;
938
939        // Handle out-of-order packets
940        self.immediate_ack_required |=
941            self.is_out_of_order(packet_number, prev_largest_ack_eliciting, dedup);
942
943        // Arm max_ack_delay timer if necessary
944        if self.earliest_ack_eliciting_since_last_ack_sent.is_none() && !self.can_send() {
945            self.earliest_ack_eliciting_since_last_ack_sent = Some(now);
946            return true;
947        }
948
949        false
950    }
951
952    fn is_out_of_order(
953        &self,
954        packet_number: u64,
955        prev_largest_ack_eliciting: u64,
956        dedup: &Dedup,
957    ) -> bool {
958        match self.reordering_threshold {
959            0 => false,
960            1 => {
961                // From https://www.rfc-editor.org/rfc/rfc9000#section-13.2.1-7
962                packet_number < prev_largest_ack_eliciting
963                    || dedup.missing_in_interval(prev_largest_ack_eliciting, packet_number)
964            }
965            _ => {
966                // From acknowledgement frequency draft, section 6.1: send an ACK immediately if
967                // doing so would cause the sender to detect a new packet loss
968                let Some((largest_acked, largest_unacked)) =
969                    self.largest_acked.zip(self.largest_ack_eliciting_packet)
970                else {
971                    return false;
972                };
973                if self.reordering_threshold > largest_acked {
974                    return false;
975                }
976                // The largest packet number that could be declared lost without a new ACK being
977                // sent
978                let largest_reported = largest_acked - self.reordering_threshold + 1;
979                let Some(smallest_missing_unreported) =
980                    dedup.smallest_missing_in_interval(largest_reported, largest_unacked)
981                else {
982                    return false;
983                };
984                largest_unacked - smallest_missing_unreported >= self.reordering_threshold
985            }
986        }
987    }
988
989    /// Should be called whenever ACKs have been sent
990    ///
991    /// This will suppress sending further ACKs until additional ACK eliciting frames arrive
992    pub(super) fn acks_sent(&mut self) {
993        // It is possible (though unlikely) that the ACKs we just sent do not cover all the
994        // ACK-eliciting packets we have received (e.g. if there is not enough room in the packet to
995        // fit all the ranges). To keep things simple, however, we assume they do. If there are
996        // indeed some ACKs that weren't covered, the packets might be ACKed later anyway, because
997        // they are still contained in `self.ranges`. If we somehow fail to send the ACKs at a later
998        // moment, the peer will assume the packets got lost and will retransmit their frames in a
999        // new packet, which is suboptimal, because we already received them. Our assumption here is
1000        // that simplicity results in code that is more performant, even in the presence of
1001        // occasional redundant retransmits.
1002        self.immediate_ack_required = false;
1003        self.ack_eliciting_since_last_ack_sent = 0;
1004        self.non_ack_eliciting_since_last_ack_sent = 0;
1005        self.earliest_ack_eliciting_since_last_ack_sent = None;
1006        self.largest_acked = self.largest_ack_eliciting_packet;
1007    }
1008
1009    /// Insert one packet that needs to be acknowledged
1010    pub(super) fn insert_one(&mut self, packet: u64, now: Instant) {
1011        self.ranges.insert_one(packet);
1012
1013        if self.largest_packet.is_none_or(|(pn, _)| packet > pn) {
1014            self.largest_packet = Some((packet, now));
1015        }
1016
1017        if self.ranges.len() > MAX_ACK_BLOCKS {
1018            self.ranges.pop_min();
1019        }
1020    }
1021
1022    /// Remove ACKs of packets numbered at or below `max` from the set of pending ACKs
1023    pub(super) fn subtract_below(&mut self, max: u64) {
1024        self.ranges.remove(0..(max + 1));
1025    }
1026
1027    /// Returns the set of currently pending ACK ranges
1028    pub(super) fn ranges(&self) -> &ArrayRangeSet {
1029        &self.ranges
1030    }
1031
1032    /// Queue an ACK if a significant number of non-ACK-eliciting packets have not yet been
1033    /// acknowledged
1034    ///
1035    /// Should be called immediately before a non-probing packet is composed, when we've already
1036    /// committed to sending a packet regardless.
1037    pub(super) fn maybe_ack_non_eliciting(&mut self) {
1038        // If we're going to send a packet anyway, and we've received a significant number of
1039        // non-ACK-eliciting packets, then include an ACK to help the peer perform timely loss
1040        // detection even if they're not sending any ACK-eliciting packets themselves. Exact
1041        // threshold chosen somewhat arbitrarily.
1042        const LAZY_ACK_THRESHOLD: u64 = 10;
1043        if self.non_ack_eliciting_since_last_ack_sent > LAZY_ACK_THRESHOLD {
1044            self.immediate_ack_required = true;
1045        }
1046    }
1047}

1049/// Helper for mitigating [optimistic ACK attacks]
1050///
1051/// A malicious peer could prompt the local application to begin a large data transfer, and then
1052/// send ACKs without first waiting for data to be received. This could defeat congestion control,
1053/// allowing the connection to consume disproportionate resources. We therefore occasionally skip
1054/// packet numbers, and classify any ACK referencing a skipped packet number as a transport error.
1055///
1056/// Skipped packet numbers occur only in the application data space (where costly transfers might
1057/// take place) and are distributed exponentially to reflect the reduced likelihood and impact of
1058/// bad behavior from a peer that has been well-behaved for an extended period.
1059///
1060/// ACKs for packet numbers that have not yet been allocated are also a transport error, but an
1061/// attacker with knowledge of the congestion control algorithm in use could time falsified ACKs to
1062/// arrive after the packets they reference are sent.
1063///
1064/// [optimistic ACK attacks]: https://www.rfc-editor.org/rfc/rfc9000.html#name-optimistic-ack-attack
1065pub(super) struct PacketNumberFilter {
1066    /// Next outgoing packet number to skip
1067    next_skipped_packet_number: u64,
1068    /// Most recently skipped packet number
1069    prev_skipped_packet_number: Option<u64>,
1070    /// Next packet number to skip is randomly selected from 2^n..2^n+1
1071    exponent: u32,
1072}
1073
1074impl PacketNumberFilter {
1075    pub(super) fn new(rng: &mut (impl Rng + ?Sized)) -> Self {
1076        // First skipped PN is in 0..64
1077        let exponent = 6;
1078        Self {
1079            next_skipped_packet_number: rng.random_range(0..2u64.saturating_pow(exponent)),
1080            prev_skipped_packet_number: None,
1081            exponent,
1082        }
1083    }
1084
1085    #[cfg(test)]
1086    pub(super) fn disabled() -> Self {
1087        Self {
1088            next_skipped_packet_number: u64::MAX,
1089            prev_skipped_packet_number: None,
1090            exponent: u32::MAX,
1091        }
1092    }
1093
1094    /// Whether to use the provided packet number (false) or to skip it (true)
1095    pub(super) fn skip_pn(&mut self, n: u64, rng: &mut (impl Rng + ?Sized)) -> bool {
1096        if n != self.next_skipped_packet_number {
1097            return false;
1098        }
1099
1100        trace!("skipping pn {n}");
1101        // Skip this packet number, and choose the next one to skip
1102        self.prev_skipped_packet_number = Some(self.next_skipped_packet_number);
1103        let next_exponent = self.exponent.saturating_add(1);
1104        self.next_skipped_packet_number = rng
1105            .random_range(2u64.saturating_pow(self.exponent)..2u64.saturating_pow(next_exponent));
1106        self.exponent = next_exponent;
1107        true
1108    }
1109}

1278        assert_eq!(acks.ack_delay(t2), Duration::from_millis(0));
1279        assert_eq!(acks.ack_delay(t3), Duration::from_millis(0));
1280
1281        // In-order packet
1282        dedup.insert(0);
1283        acks.insert_one(0, t1);
1284        acks.packet_received(t1, 0, true, &dedup);
1285        assert_eq!(acks.ack_delay(t1), Duration::from_millis(0));
1286        assert_eq!(acks.ack_delay(t2), Duration::from_millis(2));
1287        assert_eq!(acks.ack_delay(t3), Duration::from_millis(7));
1288
1289        // Out of order (higher than expected)
1290        dedup.insert(3);
1291        acks.insert_one(3, t2);
1292        acks.packet_received(t2, 3, true, &dedup);
1293        assert_eq!(acks.ack_delay(t2), Duration::from_millis(0));
1294        assert_eq!(acks.ack_delay(t3), Duration::from_millis(5));
1295
1296        // Out of order (lower than expected, so previous instant is kept)
1297        dedup.insert(2);
1298        acks.insert_one(2, t3);
1299        acks.packet_received(t3, 2, true, &dedup);
1300        assert_eq!(acks.ack_delay(t3), Duration::from_millis(5));
