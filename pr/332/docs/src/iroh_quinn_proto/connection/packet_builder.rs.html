14/// QUIC packet builder
15///
16/// This allows building QUIC packets: it takes care of writing the header, allows writing
17/// frames and on [`PacketBuilder::finish`] (or [`PacketBuilder::finish_and_track`]) it
18/// encrypts the packet so it is ready to be sent on the wire.
19///
20/// The builder manages the write buffer into which the packet is written, and directly
21/// implements [`BufMut`] to write frames into the packet.
22pub(super) struct PacketBuilder<'a, 'b> {
23    pub(super) buf: &'a mut TransmitBuf<'b>,
24    pub(super) space: SpaceId,
25    path: PathId,
26    pub(super) partial_encode: PartialEncode,
27    pub(super) ack_eliciting: bool,
28    pub(super) exact_number: u64,
29    pub(super) short_header: bool,
30    /// Smallest absolute position in the associated buffer that must be occupied by this packet's
31    /// frames
32    pub(super) min_size: usize,
33    pub(super) tag_len: usize,
34    pub(super) _span: tracing::span::EnteredSpan,
35    qlog: QlogSentPacket,
36    sent_frames: SentFrames,
37}

40    /// Write a new packet header to `buffer` and determine the packet's properties
41    ///
42    /// Marks the connection drained and returns `None` if the confidentiality limit would be
43    /// violated.
44    pub(super) fn new(
45        now: Instant,
46        space_id: SpaceId,
47        path_id: PathId,
48        dst_cid: ConnectionId,
49        buffer: &'a mut TransmitBuf<'b>,
50        ack_eliciting: bool,
51        conn: &mut Connection,
52    ) -> Option<Self>
53    where
54        'b: 'a,
55    {
56        let mut qlog = QlogSentPacket::default();
57
58        let version = conn.version;
59        // Initiate key update if we're approaching the confidentiality limit
60        let sent_with_keys = conn.spaces[space_id].sent_with_keys();
61        if space_id == SpaceId::Data {
62            if sent_with_keys >= conn.key_phase_size {
63                debug!("routine key update due to phase exhaustion");
64                conn.force_key_update();
65            }
66        } else {
67            let confidentiality_limit = conn.spaces[space_id]
68                .crypto
69                .as_ref()
70                .map_or_else(
71                    || &conn.zero_rtt_crypto.as_ref().unwrap().packet,
72                    |keys| &keys.packet.local,
73                )
74                .confidentiality_limit();
75            if sent_with_keys.saturating_add(1) == confidentiality_limit {
76                // We still have time to attempt a graceful close
77                conn.close_inner(
78                    now,
79                    Close::Connection(frame::ConnectionClose {
80                        error_code: TransportErrorCode::AEAD_LIMIT_REACHED,
81                        frame_type: frame::MaybeFrame::None,
82                        reason: Bytes::from_static(b"confidentiality limit reached"),
83                    }),
84                )
85            } else if sent_with_keys > confidentiality_limit {
86                // Confidentiality limited violated and there's nothing we can do
87                conn.kill(
88                    TransportError::AEAD_LIMIT_REACHED("confidentiality limit reached").into(),
89                );
90                return None;
91            }
92        }
93
94        let space = &mut conn.spaces[space_id];
95        let exact_number = space.for_path(path_id).get_tx_number(&mut conn.rng);
96        let span = trace_span!("send", space = ?space_id, pn = exact_number, %path_id).entered();
97
98        let number = PacketNumber::new(
99            exact_number,
100            space.for_path(path_id).largest_acked_packet.unwrap_or(0),
101        );
102        let header = match space_id {
103            SpaceId::Data if space.crypto.is_some() => Header::Short {
104                dst_cid,
105                number,
106                spin: if conn.spin_enabled {
107                    conn.spin
108                } else {
109                    conn.rng.random()
110                },
111                key_phase: conn.key_phase,
112            },
113            SpaceId::Data => Header::Long {
114                ty: LongType::ZeroRtt,
115                src_cid: conn.handshake_cid,
116                dst_cid,
117                number,
118                version,
119            },
120            SpaceId::Handshake => Header::Long {
121                ty: LongType::Handshake,
122                src_cid: conn.handshake_cid,
123                dst_cid,
124                number,
125                version,
126            },
127            SpaceId::Initial => Header::Initial(InitialHeader {
128                src_cid: conn.handshake_cid,
129                dst_cid,
130                token: match &conn.side {
131                    ConnectionSide::Client { token, .. } => token.clone(),
132                    ConnectionSide::Server { .. } => Bytes::new(),
133                },
134                number,
135                version,
136            }),
137        };
138
139        let partial_encode = header.encode(buffer);
140        if conn.peer_params.grease_quic_bit && conn.rng.random() {
141            buffer.as_mut_slice()[partial_encode.start] ^= FIXED_BIT;
142        }
143
144        let (sample_size, tag_len) = if let Some(ref crypto) = space.crypto {
145            (
146                crypto.header.local.sample_size(),
147                crypto.packet.local.tag_len(),
148            )
149        } else if space_id == SpaceId::Data {
150            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
151            (zero_rtt.header.sample_size(), zero_rtt.packet.tag_len())
152        } else {
153            unreachable!();
154        };
155
156        // Each packet must be large enough for header protection sampling, i.e. the combined
157        // lengths of the encoded packet number and protected payload must be at least 4 bytes
158        // longer than the sample required for header protection. Further, each packet should be at
159        // least tag_len + 6 bytes larger than the destination CID on incoming packets so that the
160        // peer may send stateless resets that are indistinguishable from regular traffic.
161
162        // pn_len + payload_len + tag_len >= sample_size + 4
163        // payload_len >= sample_size + 4 - pn_len - tag_len
164        let min_size = Ord::max(
165            buffer.len() + (sample_size + 4).saturating_sub(number.len() + tag_len),
166            partial_encode.start + dst_cid.len() + 6,
167        );
168        let max_size = buffer.datagram_max_offset() - tag_len;
169        debug_assert!(max_size >= min_size);
170
171        qlog.header(
172            &header,
173            Some(exact_number),
174            space_id,
175            space_id == SpaceId::Data && conn.spaces[SpaceId::Data].crypto.is_none(),
176            path_id,
177        );
178
179        Some(Self {
180            buf: buffer,
181            space: space_id,
182            path: path_id,
183            partial_encode,
184            exact_number,
185            short_header: header.is_short(),
186            min_size,
187            tag_len,
188            ack_eliciting,
189            qlog,
190            sent_frames: SentFrames::default(),
191            _span: span,
192        })
193    }

213    /// Append the minimum amount of padding to the packet such that, after encryption, the
214    /// enclosing datagram will occupy at least `min_size` bytes
215    pub(super) fn pad_to(&mut self, min_size: u16) {
216        // The datagram might already have a larger minimum size than the caller is requesting, if
217        // e.g. we're coalescing packets and have populated more than `min_size` bytes with packets
218        // already.
219        self.min_size = Ord::max(
220            self.min_size,
221            self.buf.datagram_start_offset() + (min_size as usize) - self.tag_len,
222        );
223    }

226    ///
227    /// It will also:
228    /// - Track the frame so that it's registered with the path once [`Self::finish_and_track`] is
229    ///   called.
230    /// - Register the sent frame with the given [`FrameStats`].
231    /// - If the qlog feature is enabled, register the frame.

234        &mut self,

238        self.write_frame_with_log_msg(frame, stats, None);

241    /// Writes a frame into the underlying buffer.
242    ///
243    /// It will also:
244    /// - Track the frame so that it's registered with the path once [`Self::finish_and_track`] is
245    ///   called.
246    /// - Register the sent frame with the given [`FrameStats`].
247    /// - If the qlog feature is enabled, register the frame.
248    /// - Log the frame. If a `msg` is given, this will be added to the log.
249    pub(super) fn write_frame_with_log_msg<'c>(
250        &mut self,
251        frame: impl Into<EncodableFrame<'c>>,
252        stats: &mut FrameStats,
253        msg: Option<&'static str>,
254    ) {
255        let frame = frame.into();
256        frame.encode(&mut self.frame_space_mut());
257        stats.record(frame.get_type());
258        self.qlog.record(&frame);
259        match msg {
260            Some(msg) => trace!(%frame, msg),
261            None => trace!(%frame),
262        }
263        self.sent_frames.record_sent_frame(frame);
264    }

279    pub(super) fn finish_and_track(
280        mut self,
281        now: Instant,
282        conn: &mut Connection,
283        path_id: PathId,
284        pad_datagram: PadDatagram,
285    ) {
286        match pad_datagram {
287            PadDatagram::No => (),
288            PadDatagram::ToSize(size) => self.pad_to(size),
289            PadDatagram::ToSegmentSize => self.pad_to(self.buf.segment_size() as u16),
290            PadDatagram::ToMinMtu => self.pad_to(MIN_INITIAL_SIZE),
291        }
292        let ack_eliciting = self.ack_eliciting;
293        let exact_number = self.exact_number;
294        let space_id = self.space;
295        let (size, padded, sent) = self.finish(conn, now);
296
297        let size = match padded || ack_eliciting {
298            true => size as u16,
299            false => 0,
300        };
301
302        let packet = SentPacket {
303            path_generation: conn.paths.get_mut(&path_id).unwrap().data.generation(),
304            largest_acked: sent.largest_acked,
305            time_sent: now,
306            size,
307            ack_eliciting,
308            retransmits: sent.retransmits,
309            stream_frames: sent.stream_frames,
310        };
311
312        conn.paths.get_mut(&path_id).unwrap().data.sent(
313            exact_number,
314            packet,
315            conn.spaces[space_id].for_path(path_id),
316        );
317        conn.reset_keep_alive(path_id, now);
318        if size != 0 {
319            if ack_eliciting {
320                conn.spaces[space_id]
321                    .for_path(path_id)
322                    .time_of_last_ack_eliciting_packet = Some(now);
323                if conn.permit_idle_reset {
324                    conn.reset_idle_timeout(now, space_id, path_id);
325                }
326                conn.permit_idle_reset = false;
327            }
328            conn.set_loss_detection_timer(now, path_id);
329            conn.path_data_mut(path_id).pacing.on_transmit(size);
330        }
331    }

333    /// Encrypt packet, returning the length of the packet and whether padding was added
334    pub(super) fn finish(
335        mut self,
336        conn: &mut Connection,
337        now: Instant,
338    ) -> (usize, bool, SentFrames) {
339        debug_assert!(
340            self.buf.len() <= self.buf.datagram_max_offset() - self.tag_len,
341            "packet exceeds maximum size"
342        );
343        let pad = self.buf.len() < self.min_size;
344        if pad {
345            let padding = self.min_size - self.buf.len();
346            trace!("PADDING * {}", padding);
347            self.buf.put_bytes(0, padding);
348            self.qlog.frame_padding(padding);
349        }
350
351        let space = &conn.spaces[self.space];
352        let (header_crypto, packet_crypto) = if let Some(ref crypto) = space.crypto {
353            (&*crypto.header.local, &*crypto.packet.local)
354        } else if self.space == SpaceId::Data {
355            let zero_rtt = conn.zero_rtt_crypto.as_ref().unwrap();
356            (&*zero_rtt.header, &*zero_rtt.packet)
357        } else {
358            unreachable!("tried to send {:?} packet without keys", self.space);
359        };
360
361        debug_assert_eq!(
362            packet_crypto.tag_len(),
363            self.tag_len,
364            "Mismatching crypto tag len"
365        );
366
367        self.buf.put_bytes(0, packet_crypto.tag_len());
368        let encode_start = self.partial_encode.start;
369        let packet_buf = &mut self.buf.as_mut_slice()[encode_start..];
370        // for packet protection, PathId::ZERO and no path are equivalent.
371        self.partial_encode.finish(
372            packet_buf,
373            header_crypto,
374            Some((self.exact_number, self.path, packet_crypto)),
375        );
376
377        let packet_len = self.buf.len() - encode_start;
378        trace!(size = %packet_len, short_header = %self.short_header, "wrote packet");
379        self.qlog.finalize(packet_len);
380        conn.qlog.emit_packet_sent(self.qlog, now);
381        (packet_len, pad, self.sent_frames)
382    }

384    /// The number of additional bytes the current packet would take up if it was finished now
385    ///
386    /// This will include any padding which is required to make the size large enough to be
387    /// encrypted correctly.
388    pub(super) fn predict_packet_end(&self) -> usize {
389        self.buf.len().max(self.min_size) + self.tag_len - self.buf.len()
390    }

392    /// Returns the remaining space in the packet that can be taken up by QUIC frames
393    ///
394    /// This leaves space in the datagram for the cryptographic tag that needs to be written
395    /// when the packet is finished.
396    pub(super) fn frame_space_remaining(&self) -> usize {
397        let max_offset = self.buf.datagram_max_offset() - self.tag_len;
398        max_offset.saturating_sub(self.buf.len())
399    }

411pub(super) enum PadDatagram {
412    /// Do not pad the datagram
413    No,
414    /// To a specific size
415    ToSize(u16),
416    /// Pad to the current MTU/segment size
417    ///
418    /// For the first datagram in a transmit the MTU is the same as the
419    /// [`TransmitBuf::segment_size`].
420    ToSegmentSize,
421    /// Pad to [`MIN_INITIAL_SIZE`], the minimal QUIC MTU of 1200 bytes
422    ToMinMtu,
423}
424
425impl std::ops::BitOrAssign for PadDatagram {
426    fn bitor_assign(&mut self, rhs: Self) {
427        *self = *self | rhs;
428    }
429}
430
431impl std::ops::BitOr for PadDatagram {
432    type Output = Self;
433
434    fn bitor(self, rhs: Self) -> Self::Output {
435        match (self, rhs) {
436            (Self::No, rhs) => rhs,
437            (Self::ToSize(size), Self::No) => Self::ToSize(size),
438            (Self::ToSize(a), Self::ToSize(b)) => Self::ToSize(a.max(b)),
439            (Self::ToSize(_), Self::ToSegmentSize) => Self::ToSegmentSize,
440            (Self::ToSize(_), Self::ToMinMtu) => Self::ToMinMtu,
441            (Self::ToSegmentSize, Self::No) => Self::ToSegmentSize,
442            (Self::ToSegmentSize, Self::ToSize(_)) => Self::ToSegmentSize,
443            (Self::ToSegmentSize, Self::ToSegmentSize) => Self::ToSegmentSize,
444            (Self::ToSegmentSize, Self::ToMinMtu) => Self::ToMinMtu,
445            (Self::ToMinMtu, _) => Self::ToMinMtu,
446        }
447    }
