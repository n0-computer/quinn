33/// Parameters governing the core QUIC state machine
34///
35/// Default values should be suitable for most internet applications. Applications protocols which
36/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
37/// `max_concurrent_uni_streams` to zero.
38///
39/// In some cases, performance or resource requirements can be improved by tuning these values to
40/// suit a particular application and/or network connection. In particular, data window sizes can be
41/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
42/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
43/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
44/// link with a 100ms round trip time.
45#[derive(Clone)]
46pub struct TransportConfig {

110    /// Maximum duration of inactivity to accept before timing out the connection.
111    ///
112    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`
113    /// represents an infinite timeout. Defaults to 30 seconds.
114    ///
115    /// **WARNING**: If a peer or its network path malfunctions or acts maliciously, an infinite
116    /// idle timeout can result in permanently hung futures!
117    ///
118    /// ```
119    /// # use std::{convert::TryInto, time::Duration};
120    /// # use iroh_quinn_proto::{TransportConfig, VarInt, VarIntBoundsExceeded};
121    /// # fn main() -> Result<(), VarIntBoundsExceeded> {
122    /// let mut config = TransportConfig::default();
123    ///
124    /// // Set the idle timeout as `VarInt`-encoded milliseconds
125    /// config.max_idle_timeout(Some(VarInt::from_u32(10_000).into()));
126    ///
127    /// // Set the idle timeout as a `Duration`
128    /// config.max_idle_timeout(Some(Duration::from_secs(10).try_into()?));
129    /// # Ok(())
130    /// # }
131    /// ```
132    pub fn max_idle_timeout(&mut self, value: Option<IdleTimeout>) -> &mut Self {
133        self.max_idle_timeout = value.map(|t| t.0);
134        self

137    /// Maximum number of bytes the peer may transmit without acknowledgement on any one stream
138    /// before becoming blocked.
139    ///
140    /// This should be set to at least the expected connection latency multiplied by the maximum
141    /// desired throughput. Setting this smaller than `receive_window` helps ensure that a single

163    /// Provides an upper bound on memory when communicating with peers that issue large amounts of

172    /// Whether to implement fair queuing for send streams having the same priority.

207    /// The initial value to be used as the maximum UDP payload size before running MTU discovery
208    /// (see [`TransportConfig::mtu_discovery_config`]).
209    ///

214    /// it down to [`TransportConfig::min_mtu`].

224    /// The maximum UDP payload size guaranteed to be supported by the network.
225    ///
226    /// Must be at least 1200, which is the default, and lower than or equal to
227    /// [`TransportConfig::initial_mtu`].
228    ///
229    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
230    /// outside of either endpoint's control. Extreme care should be used when raising this value
231    /// outside of private networks where these factors are fully controlled. If the provided value
232    /// is higher than what the network path actually supports, the result will be unpredictable and
233    /// catastrophic packet loss, without a possibility of repair. Prefer
234    /// [`TransportConfig::initial_mtu`] together with
235    /// [`TransportConfig::mtu_discovery_config`] to set a maximum UDP payload size that robustly
236    /// adapts to the network.
237    pub fn min_mtu(&mut self, value: u16) -> &mut Self {
238        self.min_mtu = value.max(INITIAL_MTU);
239        self
240    }

464    /// Sets the maximum number of nat traversal addresses this endpoint allows the remote to
465    /// advertise
466    ///
467    /// Setting this to any nonzero value will enable Iroh's holepunching, loosely based in the Nat
468    /// Traversal Extension for QUIC, see
469    /// <https://www.ietf.org/archive/id/draft-seemann-quic-nat-traversal-02.html>
470    ///
471    /// This implementation expects the multipath extension to be enabled as well. if not yet
472    /// enabled via [`Self::max_concurrent_multipath_paths`], a default value of
473    /// [`DEFAULT_CONCURRENT_MULTIPATH_PATHS_WHEN_ENABLED`] will be used.
474    pub fn set_max_remote_nat_traversal_addresses(&mut self, max_addresses: u8) -> &mut Self {
475        self.max_remote_nat_traversal_addresses = NonZeroU8::new(max_addresses);
476        if max_addresses != 0 && self.max_concurrent_multipath_paths.is_none() {
477            self.max_concurrent_multipath_paths(
478                DEFAULT_CONCURRENT_MULTIPATH_PATHS_WHEN_ENABLED_.get(),
479            );
480        }
481        self
482    }

519    pub(crate) fn create_qlog_sink(
520        &self,
521        side: Side,
522        remote: SocketAddr,
523        initial_dst_cid: ConnectionId,
524        now: Instant,
525    ) -> QlogSink {
526        #[cfg(not(feature = "qlog"))]
527        let sink = {
528            let _ = (side, remote, initial_dst_cid, now);
529            QlogSink::default()
530        };
531
532        #[cfg(feature = "qlog")]
533        let sink = {
534            if let Some(config) = self
535                .qlog_factory
536                .as_ref()
537                .and_then(|factory| factory.for_connection(side, remote, initial_dst_cid, now))
538            {
539                QlogSink::new(config, initial_dst_cid, side, now)
540            } else {
541                QlogSink::default()
542            }
543        };
544
545        sink
546    }

549impl Default for TransportConfig {
550    fn default() -> Self {
551        const EXPECTED_RTT: u32 = 100; // ms
552        const MAX_STREAM_BANDWIDTH: u32 = 12500 * 1000; // bytes/s
553        // Window size needed to avoid pipeline
554        // stalls
555        const STREAM_RWND: u32 = MAX_STREAM_BANDWIDTH / 1000 * EXPECTED_RTT;
556
557        Self {
558            max_concurrent_bidi_streams: 100u32.into(),
559            max_concurrent_uni_streams: 100u32.into(),
560            // 30 second default recommended by RFC 9308  3.2
561            max_idle_timeout: Some(VarInt(30_000)),
562            stream_receive_window: STREAM_RWND.into(),
563            receive_window: VarInt::MAX,
564            send_window: (8 * STREAM_RWND).into(),
565            send_fairness: true,
566
567            packet_threshold: 3,
568            time_threshold: 9.0 / 8.0,
569            initial_rtt: Duration::from_millis(250),
570            initial_mtu: INITIAL_MTU,
571            min_mtu: INITIAL_MTU,
572            mtu_discovery_config: Some(MtuDiscoveryConfig::default()),
573            pad_to_mtu: false,
574            ack_frequency_config: None,
575
576            persistent_congestion_threshold: 3,
577            keep_alive_interval: None,
578            crypto_buffer_size: 16 * 1024,
579            allow_spin: true,
580            datagram_receive_buffer_size: Some(STREAM_RWND as usize),
581            datagram_send_buffer_size: 1024 * 1024,
582            #[cfg(test)]
583            deterministic_packet_numbers: false,
584
585            congestion_controller_factory: Arc::new(congestion::CubicConfig::default()),
586
587            enable_segmentation_offload: true,

696/// Parameters for controlling the peer's acknowledgement frequency
697///
698/// The parameters provided in this config will be sent to the peer at the beginning of the
699/// connection, so it can take them into account when sending acknowledgements (see each parameter's
700/// description for details on how it influences acknowledgement frequency).
701///
702/// Quinn's implementation follows the fourth draft of the
703/// [QUIC Acknowledgement Frequency extension](https://datatracker.ietf.org/doc/html/draft-ietf-quic-ack-frequency-04).
704/// The defaults produce behavior slightly different than the behavior without this extension,
705/// because they change the way reordered packets are handled (see
706/// [`AckFrequencyConfig::reordering_threshold`] for details).
707#[derive(Clone, Debug)]
708pub struct AckFrequencyConfig {
709    pub(crate) ack_eliciting_threshold: VarInt,
710    pub(crate) max_ack_delay: Option<Duration>,
711    pub(crate) reordering_threshold: VarInt,
712}
713
714impl AckFrequencyConfig {
715    /// The ack-eliciting threshold we will request the peer to use
716    ///
717    /// This threshold represents the number of ack-eliciting packets an endpoint may receive
718    /// without immediately sending an ACK.
719    ///
720    /// The remote peer should send at least one ACK frame when more than this number of
721    /// ack-eliciting packets have been received. A value of 0 results in a receiver immediately
722    /// acknowledging every ack-eliciting packet.
723    ///
724    /// Defaults to 1, which sends ACK frames for every other ack-eliciting packet.
725    pub fn ack_eliciting_threshold(&mut self, value: VarInt) -> &mut Self {
726        self.ack_eliciting_threshold = value;
727        self
728    }
729
730    /// The `max_ack_delay` we will request the peer to use
731    ///
732    /// This parameter represents the maximum amount of time that an endpoint waits before sending
733    /// an ACK when the ack-eliciting threshold hasn't been reached.
734    ///
735    /// The effective `max_ack_delay` will be clamped to be at least the peer's `min_ack_delay`
736    /// transport parameter, and at most the greater of the current path RTT or 25ms.
737    ///
738    /// Defaults to `None`, in which case the peer's original `max_ack_delay` will be used, as
739    /// obtained from its transport parameters.
740    pub fn max_ack_delay(&mut self, value: Option<Duration>) -> &mut Self {
741        self.max_ack_delay = value;
742        self
743    }
744
745    /// The reordering threshold we will request the peer to use
746    ///
747    /// This threshold represents the amount of out-of-order packets that will trigger an endpoint
748    /// to send an ACK, without waiting for `ack_eliciting_threshold` to be exceeded or for
749    /// `max_ack_delay` to be elapsed.
750    ///
751    /// A value of 0 indicates out-of-order packets do not elicit an immediate ACK. A value of 1
752    /// immediately acknowledges any packets that are received out of order (this is also the
753    /// behavior when the extension is disabled).
754    ///
755    /// It is recommended to set this value to [`TransportConfig::packet_threshold`] minus one.
756    /// Since the default value for [`TransportConfig::packet_threshold`] is 3, this value defaults
757    /// to 2.
758    pub fn reordering_threshold(&mut self, value: VarInt) -> &mut Self {
759        self.reordering_threshold = value;
760        self
761    }
762}
763
764impl Default for AckFrequencyConfig {
765    fn default() -> Self {
766        Self {
767            ack_eliciting_threshold: VarInt(1),
768            max_ack_delay: None,
769            reordering_threshold: VarInt(2),
770        }
771    }
772}

774/// Parameters governing MTU discovery.
775///
776/// # The why of MTU discovery
777///
778/// By design, QUIC ensures during the handshake that the network path between the client and the
779/// server is able to transmit unfragmented UDP packets with a body of 1200 bytes. In other words,
780/// once the connection is established, we know that the network path's maximum transmission unit
781/// (MTU) is of at least 1200 bytes (plus IP and UDP headers). Because of this, a QUIC endpoint can
782/// split outgoing data in packets of 1200 bytes, with confidence that the network will be able to
783/// deliver them (if the endpoint were to send bigger packets, they could prove too big and end up
784/// being dropped).
785///
786/// There is, however, a significant overhead associated to sending a packet. If the same
787/// information can be sent in fewer packets, that results in higher throughput. The amount of
788/// packets that need to be sent is inversely proportional to the MTU: the higher the MTU, the
789/// bigger the packets that can be sent, and the fewer packets that are needed to transmit a given
790/// amount of bytes.
791///
792/// Most networks have an MTU higher than 1200. Through MTU discovery, endpoints can detect the
793/// path's MTU and, if it turns out to be higher, start sending bigger packets.
794///
795/// # MTU discovery internals
796///
797/// Quinn implements MTU discovery through DPLPMTUD (Datagram Packetization Layer Path MTU
798/// Discovery), described in [section 14.3 of RFC
799/// 9000](https://www.rfc-editor.org/rfc/rfc9000.html#section-14.3). This method consists of sending
800/// QUIC packets padded to a particular size (called PMTU probes), and waiting to see if the remote
801/// peer responds with an ACK. If an ACK is received, that means the probe arrived at the remote
802/// peer, which in turn means that the network path's MTU is of at least the packet's size. If the
803/// probe is lost, it is sent another 2 times before concluding that the MTU is lower than the
804/// packet's size.
805///
806/// MTU discovery runs on a schedule (e.g. every 600 seconds) specified through
807/// [`MtuDiscoveryConfig::interval`]. The first run happens right after the handshake, and
808/// subsequent discoveries are scheduled to run when the interval has elapsed, starting from the
809/// last time when MTU discovery completed.
810///
811/// Since the search space for MTUs is quite big (the smallest possible MTU is 1200, and the highest
812/// is 65527), Quinn performs a binary search to keep the number of probes as low as possible. The
813/// lower bound of the search is equal to [`TransportConfig::initial_mtu`] in the
814/// initial MTU discovery run, and is equal to the currently discovered MTU in subsequent runs. The
815/// upper bound is determined by the minimum of [`MtuDiscoveryConfig::upper_bound`] and the
816/// `max_udp_payload_size` transport parameter received from the peer during the handshake.
817///
818/// # Black hole detection
819///
820/// If, at some point, the network path no longer accepts packets of the detected size, packet loss
821/// will eventually trigger black hole detection and reset the detected MTU to 1200. In that case,
822/// MTU discovery will be triggered after [`MtuDiscoveryConfig::black_hole_cooldown`] (ignoring the
823/// timer that was set based on [`MtuDiscoveryConfig::interval`]).
824///
825/// # Interaction between peers
826///
827/// There is no guarantee that the MTU on the path between A and B is the same as the MTU of the
828/// path between B and A. Therefore, each peer in the connection needs to run MTU discovery
829/// independently in order to discover the path's MTU.
830#[derive(Clone, Debug)]
831pub struct MtuDiscoveryConfig {
832    pub(crate) interval: Duration,
833    pub(crate) upper_bound: u16,
834    pub(crate) minimum_change: u16,
835    pub(crate) black_hole_cooldown: Duration,
836}
837
838impl MtuDiscoveryConfig {
839    /// Specifies the time to wait after completing MTU discovery before starting a new MTU
840    /// discovery run.
841    ///
842    /// Defaults to 600 seconds, as recommended by [RFC
843    /// 8899](https://www.rfc-editor.org/rfc/rfc8899).
844    pub fn interval(&mut self, value: Duration) -> &mut Self {
845        self.interval = value;
846        self
847    }
848
849    /// Specifies the upper bound to the max UDP payload size that MTU discovery will search for.
850    ///
851    /// Defaults to 1452, to stay within Ethernet's MTU when using IPv4 and IPv6. The highest
852    /// allowed value is 65527, which corresponds to the maximum permitted UDP payload on IPv6.
853    ///
854    /// It is safe to use an arbitrarily high upper bound, regardless of the network path's MTU. The
855    /// only drawback is that MTU discovery might take more time to finish.
856    pub fn upper_bound(&mut self, value: u16) -> &mut Self {
857        self.upper_bound = value.min(MAX_UDP_PAYLOAD);
858        self
859    }
860
861    /// Specifies the amount of time that MTU discovery should wait after a black hole was detected
862    /// before running again. Defaults to one minute.
863    ///
864    /// Black hole detection can be spuriously triggered in case of congestion, so it makes sense to
865    /// try MTU discovery again after a short period of time.
866    pub fn black_hole_cooldown(&mut self, value: Duration) -> &mut Self {
867        self.black_hole_cooldown = value;
868        self
869    }
870
871    /// Specifies the minimum MTU change to stop the MTU discovery phase.
872    /// Defaults to 20.
873    pub fn minimum_change(&mut self, value: u16) -> &mut Self {
874        self.minimum_change = value;
875        self
876    }
877}
878
879impl Default for MtuDiscoveryConfig {
880    fn default() -> Self {
881        Self {
882            interval: Duration::from_secs(600),
883            upper_bound: 1452,
884            black_hole_cooldown: Duration::from_secs(60),
885            minimum_change: 20,
886        }
887    }
888}

890/// Maximum duration of inactivity to accept before timing out the connection
891///
892/// This wraps an underlying [`VarInt`], representing the duration in milliseconds. Values can be
893/// constructed by converting directly from `VarInt`, or using `TryFrom<Duration>`.
894///
895/// ```
896/// # use std::{convert::TryFrom, time::Duration};
897/// # use iroh_quinn_proto::{IdleTimeout, VarIntBoundsExceeded, VarInt};
898/// # fn main() -> Result<(), VarIntBoundsExceeded> {
899/// // A `VarInt`-encoded value in milliseconds
900/// let timeout = IdleTimeout::from(VarInt::from_u32(10_000));
901///
902/// // Try to convert a `Duration` into a `VarInt`-encoded timeout
903/// let timeout = IdleTimeout::try_from(Duration::from_secs(10))?;
904/// # Ok(())
905/// # }
906/// ```
907#[derive(Default, Copy, Clone, Eq, Hash, Ord, PartialEq, PartialOrd)]
908pub struct IdleTimeout(VarInt);
909
910impl From<VarInt> for IdleTimeout {
911    fn from(inner: VarInt) -> Self {
912        Self(inner)
913    }
914}
915
916impl std::convert::TryFrom<Duration> for IdleTimeout {
917    type Error = VarIntBoundsExceeded;
918
919    fn try_from(timeout: Duration) -> Result<Self, Self::Error> {
920        let inner = VarInt::try_from(timeout.as_millis())?;
921        Ok(Self(inner))
922    }
923}

926    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
927        self.0.fmt(f)
