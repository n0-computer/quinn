38impl Decodable for PathId {
39    fn decode<B: bytes::Buf>(r: &mut B) -> coding::Result<Self> {
40        let v = VarInt::decode(r)?;

64    /// Saturating integer addition. Computes self + rhs, saturating at the numeric bounds instead
65    /// of overflowing.
66    pub fn saturating_add(self, rhs: impl Into<Self>) -> Self {
67        let rhs = rhs.into();
68        let inner = self.0.saturating_add(rhs.0);

103/// State needed for a single path ID.
104///
105/// A single path ID can migrate according to the rules in RFC9000 §9, either voluntary or
106/// involuntary. We need to keep the [`PathData`] of the previously used such path available
107/// in order to defend against migration attacks (see RFC9000 §9.3.1, §9.3.2 and §9.3.3) as
108/// well as to support path probing (RFC9000 §9.1).
109#[derive(Debug)]
110pub(super) struct PathState {
111    pub(super) data: PathData,
112    pub(super) prev: Option<(ConnectionId, PathData)>,
113}
114
115impl PathState {
116    /// Update counters to account for a packet becoming acknowledged, lost, or abandoned
117    pub(super) fn remove_in_flight(&mut self, packet: &SentPacket) {
118        // Visit known paths from newest to oldest to find the one `pn` was sent on
119        for path_data in [&mut self.data]
120            .into_iter()
121            .chain(self.prev.as_mut().map(|(_, data)| data))
122        {
123            if path_data.remove_in_flight(packet) {
124                return;
125            }
126        }
127    }
128}

352    /// Account for transmission of `packet` with number `pn` in `space`
353    pub(super) fn sent(&mut self, pn: u64, packet: SentPacket, space: &mut PacketNumberSpace) {
354        self.in_flight.insert(&packet);
355        if self.first_packet.is_none() {
356            self.first_packet = Some(pn);
357        }
358        if let Some(forgotten) = space.sent(pn, packet) {
359            self.remove_in_flight(&forgotten);

398    pub(super) fn earliest_expiring_challenge(&self) -> Option<Instant> {
399        if self.challenges_sent.is_empty() {
400            return None;

410    /// Handle receiving a PATH_RESPONSE.
411    pub(super) fn on_path_response_received(
412        &mut self,
413        now: Instant,
414        token: u64,
415        network_path: FourTuple,
416    ) -> OnPathResponseReceived {
417        match self.challenges_sent.get(&token) {
418            // Response to an on-path PathChallenge
419            Some(info)
420                if info.network_path.is_probably_same_path(&network_path)
421                    && self.network_path.is_probably_same_path(&network_path) =>
422            {
423                self.network_path.update_local_if_same_remote(&network_path);
424                let sent_instant = info.sent_instant;
425                if !std::mem::replace(&mut self.validated, true) {
426                    trace!("new path validated");
427                }
428                // Clear any other on-path sent challenge.
429                self.challenges_sent
430                    .retain(|_token, info| !info.network_path.is_probably_same_path(&network_path));
431
432                self.send_new_challenge = false;
433
434                // This RTT can only be used for the initial RTT, not as a normal
435                // sample: https://www.rfc-editor.org/rfc/rfc9002#section-6.2.2-2.
436                let rtt = now.saturating_duration_since(sent_instant);
437                self.rtt.reset_initial_rtt(rtt);
438
439                let was_open = std::mem::replace(&mut self.open, true);
440                OnPathResponseReceived::OnPath { was_open }
441            }
442            // Response to an off-path PathChallenge
443            Some(info) if info.network_path.is_probably_same_path(&network_path) => {
444                self.challenges_sent
445                    .retain(|_token, info| !info.network_path.is_probably_same_path(&network_path));
446                OnPathResponseReceived::OffPath
447            }
448            // Response to a PathChallenge we recognize, but from an invalid remote
449            Some(info) => OnPathResponseReceived::Invalid {
450                expected: info.network_path,
451            },
452            // Response to an unknown PathChallenge
453            None => OnPathResponseReceived::Unknown,
454        }
455    }

497    /// Updates the last observed address report received on this path.
498    ///
499    /// If the address was updated, it's returned to be informed to the application.
500    #[must_use = "updated observed address must be reported to the application"]
501    pub(super) fn update_observed_addr_report(
502        &mut self,
503        observed: ObservedAddr,
504    ) -> Option<SocketAddr> {
505        match self.last_observed_addr_report.as_mut() {
506            Some(prev) => {
507                if prev.seq_no >= observed.seq_no {
508                    // frames that do not increase the sequence number on this path are ignored
509                    None
510                } else if prev.ip == observed.ip && prev.port == observed.port {
511                    // keep track of the last seq_no but do not report the address as updated
512                    prev.seq_no = observed.seq_no;
513                    None
514                } else {
515                    let addr = observed.socket_addr();
516                    self.last_observed_addr_report = Some(observed);
517                    Some(addr)
518                }
519            }
520            None => {
521                let addr = observed.socket_addr();
522                self.last_observed_addr_report = Some(observed);
523                Some(addr)
524            }
525        }
526    }

555/// Congestion metrics as described in [`recovery_metrics_updated`].
556///
557/// [`recovery_metrics_updated`]: https://datatracker.ietf.org/doc/html/draft-ietf-quic-qlog-quic-events.html#name-recovery_metrics_updated
558#[cfg(feature = "qlog")]
559#[derive(Default, Clone, PartialEq, Debug)]
560#[non_exhaustive]
561struct RecoveryMetrics {
562    pub min_rtt: Option<Duration>,
563    pub smoothed_rtt: Option<Duration>,
564    pub latest_rtt: Option<Duration>,
565    pub rtt_variance: Option<Duration>,
566    pub pto_count: Option<u32>,
567    pub bytes_in_flight: Option<u64>,
568    pub packets_in_flight: Option<u64>,
569    pub congestion_window: Option<u64>,
570    pub ssthresh: Option<u64>,
571    pub pacing_rate: Option<u64>,
572}
573
574#[cfg(feature = "qlog")]
575impl RecoveryMetrics {
576    /// Retain only values that have been updated since the last snapshot.
577    fn retain_updated(&self, previous: &Self) -> Self {
578        macro_rules! keep_if_changed {
579            ($name:ident) => {
580                if previous.$name == self.$name {
581                    None
582                } else {
583                    self.$name
584                }
585            };
586        }
587
588        Self {
589            min_rtt: keep_if_changed!(min_rtt),
590            smoothed_rtt: keep_if_changed!(smoothed_rtt),
591            latest_rtt: keep_if_changed!(latest_rtt),
592            rtt_variance: keep_if_changed!(rtt_variance),
593            pto_count: keep_if_changed!(pto_count),
594            bytes_in_flight: keep_if_changed!(bytes_in_flight),
595            packets_in_flight: keep_if_changed!(packets_in_flight),
596            congestion_window: keep_if_changed!(congestion_window),
597            ssthresh: keep_if_changed!(ssthresh),
598            pacing_rate: keep_if_changed!(pacing_rate),
599        }
600    }
601
602    /// Emit a `MetricsUpdated` event containing only updated values
603    fn to_qlog_event(&self, path_id: PathId, previous: &Self) -> Option<RecoveryMetricsUpdated> {
604        let updated = self.retain_updated(previous);
605
606        if updated == Self::default() {
607            return None;
608        }
609
610        Some(RecoveryMetricsUpdated {
611            min_rtt: updated.min_rtt.map(|rtt| rtt.as_secs_f32()),
612            smoothed_rtt: updated.smoothed_rtt.map(|rtt| rtt.as_secs_f32()),
613            latest_rtt: updated.latest_rtt.map(|rtt| rtt.as_secs_f32()),
614            rtt_variance: updated.rtt_variance.map(|rtt| rtt.as_secs_f32()),
615            pto_count: updated
616                .pto_count
617                .map(|count| count.try_into().unwrap_or(u16::MAX)),
618            bytes_in_flight: updated.bytes_in_flight,
619            packets_in_flight: updated.packets_in_flight,
620            congestion_window: updated.congestion_window,
621            ssthresh: updated.ssthresh,
622            pacing_rate: updated.pacing_rate,
623            path_id: Some(path_id.as_u32() as u64),
624        })
625    }
626}

640
641impl RttEstimator {
642    pub(super) fn new(initial_rtt: Duration) -> Self {
643        Self {
644            latest: initial_rtt,
645            smoothed: None,
646            var: initial_rtt / 2,
647            min: initial_rtt,
648        }
649    }
650
651    /// Resets the estimator using a new initial_rtt value.
652    ///
653    /// This only resets the initial_rtt **if** no samples have been recorded yet. If there
654    /// are any recorded samples the initial estimate can not be adjusted after the fact.
655    ///
656    /// This is useful when you receive a PATH_RESPONSE in the first packet received on a
657    /// new path. In this case you can use the delay of the PATH_CHALLENGE-PATH_RESPONSE as
658    /// the initial RTT to get a better expected estimation.
659    ///
660    /// A PATH_CHALLENGE-PATH_RESPONSE pair later in the connection should not be used
661    /// explicitly as an estimation since PATH_CHALLENGE is an ACK-eliciting packet itself
662    /// already.
663    pub(crate) fn reset_initial_rtt(&mut self, initial_rtt: Duration) {
664        if self.smoothed.is_none() {
665            self.latest = initial_rtt;
666            self.var = initial_rtt / 2;
667            self.min = initial_rtt;
668        }
669    }
670
671    /// The current best RTT estimation.
672    pub fn get(&self) -> Duration {
673        self.smoothed.unwrap_or(self.latest)
674    }
675
676    /// Conservative estimate of RTT
677    ///
678    /// Takes the maximum of smoothed and latest RTT, as recommended
679    /// in 6.1.2 of the recovery spec (draft 29).
680    pub fn conservative(&self) -> Duration {
681        self.get().max(self.latest)
682    }
683
684    /// Minimum RTT registered so far for this estimator.
685    pub fn min(&self) -> Duration {
686        self.min
687    }
688
689    /// PTO computed as described in RFC9002#6.2.1.
690    pub(crate) fn pto_base(&self) -> Duration {
691        self.get() + cmp::max(4 * self.var, TIMER_GRANULARITY)
692    }
693
694    /// Records an RTT sample.
695    pub(crate) fn update(&mut self, ack_delay: Duration, rtt: Duration) {
696        self.latest = rtt;
697        // https://www.rfc-editor.org/rfc/rfc9002.html#section-5.2-3:
698        // min_rtt does not adjust for ack_delay to avoid underestimating.
699        self.min = cmp::min(self.min, self.latest);
700        // Based on RFC6298.
701        if let Some(smoothed) = self.smoothed {
702            let adjusted_rtt = if self.min + ack_delay <= self.latest {
703                self.latest - ack_delay
704            } else {
705                self.latest
706            };
707            let var_sample = smoothed.abs_diff(adjusted_rtt);
708            self.var = (3 * self.var + var_sample) / 4;
709            self.smoothed = Some((7 * smoothed + adjusted_rtt) / 8);
710        } else {
711            self.smoothed = Some(self.latest);
712            self.var = self.latest / 2;
713            self.min = self.latest;
714        }
715    }
716}

718#[derive(Default, Debug)]
719pub(crate) struct PathResponses {
720    pending: Vec<PathResponse>,
721}
722
723impl PathResponses {
724    pub(crate) fn push(&mut self, packet: u64, token: u64, network_path: FourTuple) {
725        /// Arbitrary permissive limit to prevent abuse
726        const MAX_PATH_RESPONSES: usize = 16;
727        let response = PathResponse {
728            packet,
729            token,
730            network_path,
731        };
732        let existing = self
733            .pending
734            .iter_mut()
735            .find(|x| x.network_path.remote == network_path.remote);
736        if let Some(existing) = existing {
737            // Update a queued response
738            if existing.packet <= packet {
739                *existing = response;
740            }
741            return;
742        }
743        if self.pending.len() < MAX_PATH_RESPONSES {
744            self.pending.push(response);
745        } else {
746            // We don't expect to ever hit this with well-behaved peers, so we don't bother dropping
747            // older challenges.
748            trace!("ignoring excessive PATH_CHALLENGE");
749        }
750    }
751
752    pub(crate) fn pop_off_path(&mut self, network_path: FourTuple) -> Option<(u64, FourTuple)> {
753        let response = *self.pending.last()?;
754        // We use an exact comparison here, because once we've received for the first time,
755        // we really should either already have a local_ip, or we will never get one
756        // (because our OS doesn't support it).
757        if response.network_path == network_path {
758            // We don't bother searching further because we expect that the on-path response will
759            // get drained in the immediate future by a call to `pop_on_path`
760            return None;
761        }
762        self.pending.pop();
763        Some((response.token, response.network_path))
764    }
765
766    pub(crate) fn pop_on_path(&mut self, network_path: FourTuple) -> Option<u64> {
767        let response = *self.pending.last()?;
768        // Using an exact comparison. See explanation in `pop_off_path`.
769        if response.network_path != network_path {
770            // We don't bother searching further because we expect that the off-path response will
771            // get drained in the immediate future by a call to `pop_off_path`
772            return None;
773        }
774        self.pending.pop();
775        Some(response.token)
776    }
777
778    pub(crate) fn is_empty(&self) -> bool {
779        self.pending.is_empty()
780    }
781}

793/// Summary statistics of packets that have been sent on a particular path, but which have not yet
794/// been acked or deemed lost
795#[derive(Debug)]
796pub(super) struct InFlight {
797    /// Sum of the sizes of all sent packets considered "in flight" by congestion control
798    ///
799    /// The size does not include IP or UDP overhead. Packets only containing ACK frames do not
800    /// count towards this to ensure congestion control does not impede congestion feedback.
801    pub(super) bytes: u64,
802    /// Number of packets in flight containing frames other than ACK and PADDING
803    ///
804    /// This can be 0 even when bytes is not 0 because PADDING frames cause a packet to be
805    /// considered "in flight" by congestion control. However, if this is nonzero, bytes will always
806    /// also be nonzero.
807    pub(super) ack_eliciting: u64,
808}
809
810impl InFlight {
811    fn new() -> Self {
812        Self {
813            bytes: 0,
814            ack_eliciting: 0,
815        }
816    }
817
818    fn insert(&mut self, packet: &SentPacket) {
819        self.bytes += u64::from(packet.size);
820        self.ack_eliciting += u64::from(packet.ack_eliciting);
821    }
822
823    /// Update counters to account for a packet becoming acknowledged, lost, or abandoned
824    fn remove(&mut self, packet: &SentPacket) {
825        self.bytes -= u64::from(packet.size);
826        self.ack_eliciting -= u64::from(packet.ack_eliciting);
827    }
828}

830/// State for QUIC-MULTIPATH PATH_STATUS_AVAILABLE and PATH_STATUS_BACKUP frames
831#[derive(Debug, Clone, Default)]
832pub(super) struct PathStatusState {
833    /// The local status
834    local_status: PathStatus,
835    /// Local sequence number, for both PATH_STATUS_AVAILABLE and PATH_STATUS_BACKUP
836    ///
837    /// This is the number of the *next* path status frame to be sent.
838    local_seq: VarInt,
839    /// The status set by the remote
840    remote_status: Option<(VarInt, PathStatus)>,
841}
842
843impl PathStatusState {
844    /// To be called on received PATH_STATUS_AVAILABLE/PATH_STATUS_BACKUP frames
845    pub(super) fn remote_update(&mut self, status: PathStatus, seq: VarInt) {
846        if self.remote_status.is_some_and(|(curr, _)| curr >= seq) {
847            return trace!(%seq, "ignoring path status update");
848        }
849
850        let prev = self.remote_status.replace((seq, status)).map(|(_, s)| s);
851        if prev != Some(status) {
852            debug!(?status, ?seq, "remote changed path status");
853        }
854    }
855
856    /// Updates the local status
857    ///
858    /// If the local status changed, the previous value is returned
859    pub(super) fn local_update(&mut self, status: PathStatus) -> Option<PathStatus> {
860        if self.local_status == status {
861            return None;
862        }
863
864        self.local_seq = self.local_seq.saturating_add(1u8);
865        Some(std::mem::replace(&mut self.local_status, status))
866    }
867
868    pub(crate) fn seq(&self) -> VarInt {
869        self.local_seq
870    }
871}

876/// <https://quicwg.org/multipath/draft-ietf-quic-multipath.html#name-path-status-management>

880pub enum PathStatus {
881    /// Paths marked with as available will be used when scheduling packets
882    ///
883    /// If multiple paths are available, packets will be scheduled on whichever has
884    /// capacity.
885    #[default]
886    Available,
887    /// Paths marked as backup will only be used if there are no available paths

890    /// expire.
891    Backup,
892}

894/// Application events about paths
895#[derive(Debug, Clone, PartialEq, Eq)]
896pub enum PathEvent {
897    /// A new path has been opened
898    Opened {
899        /// Which path is now open
900        id: PathId,
901    },
902    /// A path has been closed
903    Closed {
904        /// Which path has been closed
905        id: PathId,
906        /// Error code supplied by the peer
907        /// See <https://www.ietf.org/archive/id/draft-ietf-quic-multipath-14.html#name-error-codes>
908        /// for a list of known errors.
909        error_code: VarInt,
910    },
911    /// All remaining state for a path has been removed
912    ///
913    /// The [`PathEvent::Closed`] would have been emitted for this path earlier.
914    Abandoned {
915        /// Which path had its state dropped
916        id: PathId,
917        /// The final path stats, they are no longer available via [`Connection::stats`]
918        ///
919        /// [`Connection::stats`]: super::Connection::stats
920        path_stats: PathStats,
921    },
922    /// Path was closed locally
923    LocallyClosed {
924        /// Path for which the error occurred
925        id: PathId,
926        /// The error that occurred
927        error: PathError,
928    },
929    /// The remote changed the status of the path
930    ///
931    /// The local status is not changed because of this event. It is up to the application
932    /// to update the local status, which is used for packet scheduling, when the remote
933    /// changes the status.
934    RemoteStatus {
935        /// Path which has changed status
936        id: PathId,
937        /// The new status set by the remote
938        status: PathStatus,
939    },
940    /// Received an observation of our external address from the peer.
941    ObservedAddr {
942        /// Path over which the observed address was reported, [`PathId::ZERO`] when multipath is
943        /// not negotiated
944        id: PathId,
945        /// The address observed by the remote over this path
946        addr: SocketAddr,
947    },
948}
