67/// Consolidated crypto state for a connection.
68///
69/// This struct groups all cryptographic state together, including:
70/// - The TLS session
71/// - Per-space keys and crypto streams
72/// - Key update state (prev/next keys)
73/// - 0-RTT state
74pub(super) struct CryptoState {
75    /// Per encryption level crypto data (Initial, Handshake, Data).
76    pub(super) spaces: [CryptoSpace; 3],
77    /// The TLS session.
78    pub(super) session: Box<dyn crypto::Session>,
79
80    /*
81     * 0-RTT related fields
82     */
83    /// Whether 0-RTT was accepted.
84    pub(super) accepted_0rtt: bool,
85    /// Whether or not 0-RTT was enabled during the handshake. Does not imply acceptance.
86    pub(super) zero_rtt_enabled: bool,
87    /// 0-RTT crypto state, cleared when no longer needed.
88    pub(super) zero_rtt_crypto: Option<ZeroRttCrypto>,
89    /// Number of packets encrypted with 0-RTT keys. Client only.
90    sent_with_zero_rtt: u64,
91
92    /*
93     * State to manage 1-RTT key updates
94     */
95    /// 1-RTT keys to be used for the next key update.
96    ///
97    /// These are generated in advance to prevent timing attacks and/or DoS by third-party
98    /// attackers spoofing key updates.
99    pub(super) next_crypto: Option<KeyPair<Box<dyn PacketKey>>>,
100    /// 1-RTT keys used prior to a key update.
101    pub(super) prev_crypto: Option<PrevCrypto>,
102    /// Current key phase, toggled on each 1-RTT key update.
103    pub(super) key_phase: bool,
104    /// How many packets are in the current key phase. Used only for `Data` space.
105    pub(super) key_phase_size: u64,
106}

108impl CryptoState {
109    pub(super) fn new(
110        session: Box<dyn crypto::Session>,
111        init_cid: ConnectionId,
112        side: Side,
113        rng: &mut impl Rng,
114    ) -> Self {
115        let initial_keys = session.initial_keys(init_cid, side);
116        let initial_space = CryptoSpace {
117            keys: Some(initial_keys),
118            ..Default::default()
119        };
120        Self {
121            spaces: [initial_space, Default::default(), Default::default()],
122            session,
123            next_crypto: None,
124            prev_crypto: None,
125            accepted_0rtt: false,
126            zero_rtt_enabled: false,
127            zero_rtt_crypto: None,
128            sent_with_zero_rtt: 0,
129            key_phase: false,
130            // A small initial key phase size ensures peers that don't handle key updates correctly
131            // fail sooner rather than later. It's okay for both peers to do this, as the first one
132            // to perform an update will reset the other's key phase size in `update_keys`, and a
133            // simultaneous key update by both is just like a regular key update with a really fast
134            // response. Inspired by quic-go's similar behavior of performing the first key update
135            // at the 100th short-header packet.
136            key_phase_size: rng.random_range(10..1000),
137        }
138    }

140    /// Removes header protection of a packet, or returns `None` if the packet was dropped.
141    pub(super) fn unprotect_header(
142        &self,
143        partial_decode: PartialDecode,
144        stateless_reset_token: Option<ResetToken>,
145    ) -> Option<UnprotectHeaderResult> {
146        let encryption_level = partial_decode.encryption_level();
147        let header_crypto = match encryption_level {
148            Some(level) => match self.remote_crypto(level) {
149                Some(crypto) => Some(crypto.0),
150                None => {
151                    let bytes = partial_decode.len();
152                    debug!(?encryption_level, bytes, "dropping unexpected packet");
153                    return None;
154                }
155            },
156            // Unprotected packet
157            None => None,
158        };
159
160        let packet = partial_decode.data();
161        let stateless_reset = packet.len() >= RESET_TOKEN_SIZE + 5
162            && stateless_reset_token.as_deref() == Some(&packet[packet.len() - RESET_TOKEN_SIZE..]);
163
164        match partial_decode.finish(header_crypto) {
165            Ok(packet) => Some(UnprotectHeaderResult {
166                packet: Some(packet),
167                stateless_reset,
168            }),
169            Err(_) if stateless_reset => Some(UnprotectHeaderResult {
170                packet: None,
171                stateless_reset: true,
172            }),
173            Err(e) => {
174                trace!("unable to complete packet decoding: {}", e);
175                None
176            }
177        }
178    }

180    /// Decrypts a packet's body in-place.
181    pub(super) fn decrypt_packet_body(
182        &self,
183        packet: &mut Packet,
184        path_id: PathId,
185        spaces: &[PacketSpace; 3],
186    ) -> Result<Option<DecryptPacketResult>, Option<TransportError>> {
187        let conn_key_phase = self.key_phase;
188        if !packet.header.is_protected() {
189            // Unprotected packets also don't have packet numbers
190            return Ok(None);
191        }
192        let space = packet.header.space();
193
194        if path_id != PathId::ZERO && space != SpaceKind::Data {
195            // do not try to decrypt illegal multipath packets
196            return Err(Some(TransportError::PROTOCOL_VIOLATION(
197                "multipath packet on non Data packet number space",
198            )));
199        }
200        // Packets that do not belong to known path ids are valid as long as they can be decrypted.
201        // If we didn't have a path, that's for the purposes of this function equivalent to not
202        // having received packets on that path yet. So both of these cases are represented by `None`.
203        let rx_packet = spaces[space]
204            .path_space(path_id)
205            .and_then(|s| s.rx_packet);
206        let number = packet
207            .header
208            .number()
209            .ok_or(None)?
210            .expand(rx_packet.map(|n| n + 1).unwrap_or_default());
211        let packet_key_phase = packet.header.key_phase();
212
213        let mut crypto_update = false;
214        let crypto = if packet.header.is_0rtt() {
215            let (_, packet) = self.remote_crypto(EncryptionLevel::ZeroRtt).unwrap();
216            packet
217        } else if packet_key_phase == conn_key_phase || space != SpaceKind::Data {
218            let (_, packet) = self.remote_crypto(space.encryption_level()).unwrap();
219            packet
220        } else if let Some(prev) = self.prev_crypto.as_ref().and_then(|crypto| {
221            // If this packet comes prior to acknowledgment of the key update by the peer,
222            if crypto.end_packet.is_none_or(|(pn, _)| number < pn) {
223                // use the previous keys.
224                Some(crypto)
225            } else {
226                // Otherwise, this must be a remotely-initiated key update, so fall through to the
227                // final case.
228                None
229            }
230        }) {
231            &*prev.crypto.remote
232        } else {
233            // We're in the Data space with a key phase mismatch and either there is no locally
234            // initiated key update or the locally initiated key update was acknowledged by a
235            // lower-numbered packet. The key phase mismatch must therefore represent a new
236            // remotely-initiated key update.
237            crypto_update = true;
238            &*self.next_crypto.as_ref().unwrap().remote
239        };
240
241        crypto
242            .decrypt(path_id, number, &packet.header_data, &mut packet.payload)
243            .map_err(|_| {
244                trace!("decryption failed with packet number {}", number);
245                None
246            })?;
247
248        if !packet.reserved_bits_valid() {
249            return Err(Some(TransportError::PROTOCOL_VIOLATION(
250                "reserved bits set",
251            )));
252        }
253
254        let mut outgoing_key_update_acked = false;
255        if let Some(ref prev) = self.prev_crypto
256            && prev.end_packet.is_none()
257            && packet_key_phase == conn_key_phase
258        {
259            outgoing_key_update_acked = true;
260        }
261
262        if crypto_update {
263            // Validate incoming key update
264            // If `rx_packet` is `None`, then either the path is entirely new, or we haven't received
265            // any packets on this path yet. In that case, having the first packet be a crypto update
266            // is fine.
267            let invalid_packet_number = rx_packet.is_some_and(|rx_packet| number <= rx_packet);
268            if invalid_packet_number || self.prev_crypto.as_ref().is_some_and(|x| x.update_unacked)
269            {
270                trace!(?number, ?rx_packet, %path_id, "crypto update failed");
271                return Err(Some(TransportError::KEY_UPDATE_ERROR("")));
272            }
273        }
274
275        Ok(Some(DecryptPacketResult {
276            number,
277            outgoing_key_update_acked,
278            incoming_key_update: crypto_update,
279        }))
280    }

349    /// Get local (sending) crypto keys and the actual encryption level for a given space.
350    ///
351    /// This method takes a [`SpaceKind`] and resolves the encryption level automatically: for the
352    /// [`SpaceKind::Data`] space on the client side, it falls back to 0-RTT keys when 1-RTT keys
353    /// are not yet available. Resolving the appropriate encryption keys makes this method
354    /// preferable to [`Self::local_crypto`] in general.
355    ///
356    /// Returns `None` if no keys are available.
357    pub(super) fn encryption_keys(
358        &self,
359        kind: SpaceKind,
360        side: Side,
361    ) -> Option<(&dyn HeaderKey, &dyn PacketKey, EncryptionLevel)> {
362        let mut keys = self.spaces[kind].keys.as_ref().map(Keys::local);
363        let mut level = match kind {
364            SpaceKind::Initial => EncryptionLevel::Initial,
365            SpaceKind::Handshake => EncryptionLevel::Handshake,
366            SpaceKind::Data => EncryptionLevel::OneRtt,
367        };
368
369        // Clients use 0-RTT keys if 1-RTT keys are not available. Servers never encrypt 0-RTT
370        if keys.is_none() && kind == SpaceKind::Data && side.is_client() {
371            keys = self.zero_rtt_crypto.as_ref().map(ZeroRttCrypto::keys);
372            level = EncryptionLevel::ZeroRtt;
373        }
374
375        keys.map(|(header_keys, packet_keys)| (header_keys, packet_keys, level))
376    }

378    /// Perform a 1-RTT key update.
379    ///
380    /// Generates the next set of keys, rotates current keys into previous, and installs the new
381    /// keys. Updates `key_phase` and `key_phase_size` accordingly.
382    ///
383    /// PANICS: If 1-RTT keys are missing.
384    pub(super) fn update_keys(&mut self, end_packet: Option<(u64, Instant)>, remote: bool) {
385        trace!("executing key update");
386
387        let new = self
388            .session
389            .next_1rtt_keys()
390            .expect("only called for `Data` packets");
391        let confidentiality_limit = new.local.confidentiality_limit();
392        let old = mem::replace(
393            &mut self.spaces[SpaceKind::Data]
394                .keys
395                .as_mut()
396                .unwrap() // safe because update_keys() can only be triggered by short packets
397                .packet,
398            mem::replace(self.next_crypto.as_mut().unwrap(), new),
399        );
400        self.prev_crypto = Some(PrevCrypto {
401            crypto: old,
402            end_packet,
403            update_unacked: remote,
404        });
405
406        self.key_phase_size = confidentiality_limit.saturating_sub(KEY_UPDATE_MARGIN);
407        self.key_phase = !self.key_phase;
408        self.spaces[2].sent_with_keys = 0;
409    }

411    /// Number of packets encrypted with the current set of keys at `level`.
412    ///
413    /// For [`EncryptionLevel::OneRtt`], this counter resets to zero on every key update (see
414    /// [`Self::update_keys`]).
415    pub(crate) fn sent_with_keys(&self, level: EncryptionLevel) -> u64 {
416        match level {
417            EncryptionLevel::Initial => self.spaces[0].sent_with_keys,
418            EncryptionLevel::ZeroRtt => self.sent_with_zero_rtt,
419            EncryptionLevel::Handshake => self.spaces[1].sent_with_keys,
420            EncryptionLevel::OneRtt => self.spaces[2].sent_with_keys,
421        }
422    }
423
424    /// Number of packets that may still be sent before the AEAD confidentiality limit is reached
425    /// at the given encryption level.
426    ///
427    /// For [`EncryptionLevel::OneRtt`] the effective limit is the minimum of the AEAD
428    /// confidentiality limit and the current key-phase size. For all other levels the raw AEAD
429    /// confidentiality limit is used.
430    ///
431    /// Returns `None` when no keys are available for `level`.
432    pub(crate) fn remaining_packet_budget(&self, level: EncryptionLevel) -> Option<u64> {
433        let sent_with_keys = self.sent_with_keys(level);
434        let (_header_keys, packet_keys) = self.local_crypto(level)?;
435        let limit = match level {
436            EncryptionLevel::OneRtt => self.key_phase_size.min(packet_keys.confidentiality_limit()),
437            _ => packet_keys.confidentiality_limit(),
438        };
439
440        Some(limit.saturating_sub(sent_with_keys))
441    }
442
443    /// Record that a packet has been encrypted at the given level.
444    pub(crate) fn inc_sent_with_keys(&mut self, level: EncryptionLevel) {
445        let count = match level {
446            EncryptionLevel::Initial => &mut self.spaces[0].sent_with_keys,
447            EncryptionLevel::ZeroRtt => &mut self.sent_with_zero_rtt,
448            EncryptionLevel::Handshake => &mut self.spaces[1].sent_with_keys,
449            EncryptionLevel::OneRtt => &mut self.spaces[2].sent_with_keys,
450        };
451        *count = count.saturating_add(1u64);
452    }

468/// QUIC packet protection levels (RFC 9001).
469#[derive(Debug, Copy, Clone, Eq, PartialEq, Ord, PartialOrd, Hash)]
470pub(crate) enum EncryptionLevel {
471    /// Initial packets (client and server).
472    Initial,
473    /// Early data (0-RTT), client only.
474    ZeroRtt,
475    /// Handshake packets.
476    Handshake,
477    /// Application data (1-RTT).
478    OneRtt,
479}
